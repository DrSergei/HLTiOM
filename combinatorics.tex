\chapter{Комбинаторика}

\section{Принцип Дирихле и экстремальный принцип}

\subsection{Усиленный принцип Дирихле}
$n$ объектов в $k$ ящиков $\Rightarrow$ есть ящик с $\ge\lceil n/k\rceil$ объектами. Усреднение: $\max a_i\ge S/n$, $\min a_i\le S/n$.

\subsection{Экстремальный принцип}
На конечном множестве конфигураций существуют минимизирующая и максимизирующая заданную функцию. Выбрать в классе плохих конфигурацию с минимальным/максимальным значением, локально варьировать $\to$ противоречие.

\subsection{Теорема Эрдеша--Секереша}
Последовательность из $N\ge (r-1)(s-1)+1$ различных чисел содержит возрастающую длины $r$ или убывающую длины $s$. Граница точна.

\subsection{Теоремы Дилуорта и Мирски}
В конечном poset: минимальное число цепей, покрывающих $P$, равно ширине $w(P)$ (Дилуорт). Минимальное число антицепей, покрывающих $P$, равно высоте $h(P)$ (Мирски). Следствие: $|P|\ge ab+1$ $\Rightarrow$ цепь длины $a+1$ или антицепь размера $b+1$.

\subsection{Теорема Эрдеша--Ко--Радо}
$\mathcal{F}\subseteq\binom{[n]}{k}$ --- пересекающееся семейство, $n\ge 2k$ $\Rightarrow$ $|\mathcal{F}|\le\binom{n-1}{k-1}$. Равенство на звезде (все $k$-множества, содержащие фиксированный элемент).

\subsection{Теорема Шура}
$N\ge S(r)$ $\Rightarrow$ любая $r$-раскраска $\{1,\dots,N\}$ содержит одноцветную $x+y=z$.

\subsection{Теорема Рамсея (биномиальная оценка)}
$R(s,t)\le R(s-1,t)+R(s,t-1)$, откуда $R(s,t)\le\binom{s+t-2}{s-1}$ и $R(s,s)\le 4^{s-1}$.

\subsection{Неравенство Боллобаша}
Пары $(A_i,B_i)$ с $A_i\cap B_i=\emptyset$ и $A_i\cap B_j\ne\emptyset$ при $i\ne j$: $\sum_i\binom{|A_i|+|B_i|}{|A_i|}^{-1}\le 1$.

\section{Вероятностный метод}

\subsection{Линейность ожидания}
$\mathbb{E}[\sum X_i]=\sum\mathbb{E}[X_i]$ для любых (зависимых) интегрируемых $X_i$.

\subsection{Метод первого момента}
$X\ge 0$, $\mathbb{E}[X]<1$ $\Rightarrow$ $P(X=0)>0$ (существование).

\subsection{Метод изменения/удаления (alteration)}
Случайная конструкция $C$ почти хороша. Удаляем дефекты. Получаем чистую конструкцию размера $\ge\mathbb{E}[|C|]-\mathbb{E}[Y]$, где $Y$ --- число дефектов.

\subsection{Локальная лемма Ловаса (LLL)}
События $A_i$, $P(A_i)\le p$, каждое зависит от $\le d$ других, $ep(d+1)\le 1$ $\Rightarrow$ $P(\bigcap\overline{A_i})>0$.

\subsection{Метод второго момента}
$P(X=0)\le\frac{\mathrm{Var}(X)}{(\mathbb{E}[X])^2}$. Если $\mathrm{Var}(X)=o((\mathbb{E}[X])^2)$, то $X>0$ с вероятностью $\to 1$.

\subsection{Случайная нижняя оценка чисел Рамсея}
$R(k,k) > 2^{k/2}$ для $k\ge 3$. Улучшение удалением: $\frac{k}{e\sqrt{2}}2^{k/2}$.

\subsection{Неравенство Чернова--Хеффдинга}
$X_i\in[a_i,b_i]$ независимы, $S=\sum X_i$: $P(|S-\mathbb{E}[S]|\ge t)\le 2\exp(-2t^2/\sum(b_i-a_i)^2)$.

\section{Биективные аргументы и двойной счёт}

\subsection{Двойной счёт (матрица инцидентности)}
Для $R\subseteq X\times Y$: $|R|=\sum_x|\{y\}|=\sum_y|\{x\}|$. Следствия: $\sum\deg v=2|E|$, тождество Вандермонда, хоккейная клюшка.

\subsection{Инволюция со сменой знака}
$\iota:X\to X$ --- инволюция, меняющая знак вне неподвижных точек $\Rightarrow$ $|X^+|-|X^-|=|\mathrm{Fix}\cap X^+|-|\mathrm{Fix}\cap X^-|$.

\subsection{Неравенство LYM и теорема Шпернера}
$\mathcal{F}\subseteq 2^{[n]}$ --- антицепь $\Rightarrow$ $\sum_{F\in\mathcal{F}}1/\binom{n}{|F|}\le 1$. Следствие: $|\mathcal{F}|\le\binom{n}{\lfloor n/2\rfloor}$.

\subsection{Лемма Бёрнсайда}
Конечная группа $G$ действует на $X$ $\Rightarrow$ число орбит $=\frac{1}{|G|}\sum_{g\in G}|X^g|$.

\subsection{Теорема Холла (о паросочетаниях)}
$A_1,\dots,A_n$ имеют СДР $\iff$ для всех $S\subseteq[n]$: $|\bigcup_{i\in S}A_i|\ge|S|$.

\subsection{Формула Кэли и код Прюфера}
Число помеченных деревьев на $n$ вершинах $=n^{n-2}$.

\subsection{Циклическая лемма (Дворецкий--Моцкин)}
В круговой последовательности из $a$ единиц и $b$ минус единиц с $a-b=k>0$ ровно $k$ сдвигов дают все частичные суммы строго положительными.

\subsection{Лемма Линдстрёма--Гесселя--Вьенно (LGV)}
В ациклическом орграфе $\det[e(A_i,B_j)]=\sum_{\text{непересек. пути}}\operatorname{sgn}(\sigma)\prod w(P_i)$, где $e(A_i,B_j)$ --- сумма весов путей.

\medskip

\section{Линейно-алгебраическая комбинаторика}

\subsection{Аргумент размерности}
$v_1,\dots,v_m$ линейно независимы в $V$ размерности $n$ $\Rightarrow$ $m\le n$.

\subsection{Диагональный признак независимости}
Если $f_i(x_j)\ne 0$ при $i=j$ и $=0$ при $i\ne j$, то $f_i$ линейно независимы.

\subsection{Oddtown (нечётный город)}
$A_i\subseteq[n]$, $|A_i|$ нечётны, $|A_i\cap A_j|$ чётны при $i\ne j$ $\Rightarrow$ $m\le n$.

\subsection{Eventown (чётный город)}
$|A_i|$ и $|A_i\cap A_j|$ чётны для всех $i,j$ $\Rightarrow$ $m\le 2^{\lfloor n/2\rfloor}$.

\subsection{Неравенство Фишера}
$A_i\subseteq[n]$, $|A_i\cap A_j|=\lambda\ge 1$ при $i\ne j$ $\Rightarrow$ $m\le n$.

\subsection{Теорема Рэя-Чаудхури--Уилсона}
$\mathcal{F}\subseteq\binom{[n]}{k}$, $|A_i\cap A_j|\in L$, $|L|=s$, $k\notin L$ $\Rightarrow$ $m\le\binom{n}{s}$.

\subsection{Комбинаторный Nullstellensatz (Алон)}
$f\in\mathbb{F}[x_1,\dots,x_n]$ степени $d=\sum t_i$, коэффициент при $\prod x_i^{t_i}\ne 0$. $S_i\subseteq\mathbb{F}$, $|S_i|>t_i$ $\Rightarrow$ $\exists s\in\prod S_i$ с $f(s)\ne 0$.

\subsection{Множества двух расстояний (Блокхёйс)}
$S\subset\mathbb{R}^n$ с ровно двумя значениями попарных расстояний $\Rightarrow$ $|S|\le\binom{n+2}{2}$.

\section{Аддитивная комбинаторика}

\subsection{Теорема Коши--Дэвенпорта (Cauchy–Davenport)}
Для непустых $A,B\subseteq\mathbb{Z}/p\mathbb{Z}$ ($p$ простое):
$$|A+B|\ge\min(p,\,|A|+|B|-1).$$

\textit{Когда использовать.} Нижние оценки сумм, существование решений $a+b=c$ в подмножествах $\mathbb{F}_p$.

\textit{Идея.} Индукция по $|A|$: сдвиг и удаление пересечения.

\subsection{Неравенство треугольника Руцца (Ruzsa)}
Для непустых $A,B,C\subseteq G$ (абелева группа):
$$|A+C|\cdot|B|\le|A+B|\cdot|B+C|.$$

\textit{Следствие.} $|A-A|\le\bigl(|A+A|/|A|\bigr)^2|A|$.

\subsection{Лемма Петридиса}
Для $A,B\subseteq G$ с $|A+B|\le K|A|$ существует непустое $X\subseteq A$ с $|X+B'|\le K|X|$ для любого $B'\subseteq B$, где $B'$ --- произвольное подмножество.

\subsection{Неравенство Пленнеке--Руже (Plünnecke–Ruzsa)}
Если $|A+A|\le K|A|$, то $|nA-mA|\le K^{n+m}|A|$, где $nA = A+\cdots+A$ ($n$ раз).

\textit{Когда использовать.} Оценки многократных сумм при малом удвоении множества.

\subsection{Теорема Эрдёша--Гинзбурга--Зива (EGZ)}
Среди любых $2n-1$ целых чисел найдутся $n$, сумма которых кратна $n$.

\textit{Обобщение (Кемниц):} Среди $2n-1$ элементов любой абелевой группы порядка $n$ есть $n$ с суммой $0$.

\section{Производящие функции (OGF и EGF)}

\subsection{Основные операции}
OGF: $A(x)=\sum a_n x^n$: $AB$ --- свёртка, $x A$ --- сдвиг, $A/(1-x)$ --- частичные суммы.
EGF: $\hat A(x)=\sum a_n x^n/n!$: произведение даёт биномиальную свёртку.

\subsection{Фильтр по корням из единицы}
$\sum_{n\equiv r\pmod m}a_n x^n=\frac1m\sum_{j=0}^{m-1}\omega^{-jr}A(\omega^j x)$ с $\omega=e^{2\pi i/m}$.

\subsection{Линейные рекурренты с постоянными коэффициентами}
$\sum c_i a_{n+i}=0$ $\Rightarrow$ $A(x)=P(x)/Q(x)$, где $Q(x)=c_k+\dots+c_0x^k=\prod(1-\alpha_j x)^{m_j}$. Тогда $a_n=\sum Q_j(n)\alpha_j^n$.

\subsection{Числа Каталана}
$C_{n+1}=\sum C_k C_{n-k}$, $C_0=1$ $\Rightarrow$ $C(x)=1+x C(x)^2$, $C_n=\frac1{n+1}\binom{2n}{n}$.

\subsection{Формула обращения Лагранжа}
$w=x\phi(w)$, $\phi(0)\ne 0$ $\Rightarrow$ $[x^n]H(w)=\frac1n[t^{n-1}] H'(t)\phi(t)^n$.

\subsection{Экспоненциальная формула}
Помеченный класс $\mathcal{F}$ = неупорядоченное разбиение носителя на $\mathcal{C}$-блоки $\Rightarrow$ $\hat F(x)=\exp(\hat C(x))$.

\subsection{Эйлерово произведение для разбиений}
$P(q)=\sum p(n)q^n=\prod_{k\ge1}\frac1{1-q^k}$. С различными частями: $\prod(1+q^k)$. С нечётными: $\prod\frac1{1-q^{2k-1}}$.

\section{Рекуррентные соотношения и закрытые формы}

\subsection{Замкнутая форма линейной рекурренты}
$a_n=\sum c_i a_{n-i}$, корни $\lambda_i$ кратности $m_i$ $\Rightarrow$ $a_n=\sum P_i(n)\lambda_i^n$, $\deg P_i\le m_i-1$. Неоднородный случай: частное решение $n^s R(n)\mu^n$ с $s$ --- кратность $\mu$ как корня характеристического.

\subsection{Сопровождающая матрица и степени}
$\mathbf v_n=C\mathbf v_{n-1}$, $C^n$ вычисляется за $O(\log n)$ умножений.

\subsection{Числа беспорядков (derangements)}
$D_n=n!\sum_{k=0}^n(-1)^k/k!$, $D_n=nD_{n-1}+(-1)^n$. ЭПФ: $e^{-x}/(1-x)$.

\subsection{Числа Стирлинга 2-го рода}
$S(n,k)=kS(n-1,k)+S(n-1,k-1)$. $S(n,k)=\frac1{k!}\sum_{j=0}^k(-1)^j\binom{k}{j}(k-j)^n$. ЭПФ: $(e^x-1)^k/k!$.

\subsection{Тождества Фибоначчи}
$F_{n-1}F_{n+1}-F_n^2=(-1)^n$, $F_{m+n}=F_mF_{n+1}+F_{m-1}F_n$. Следствие: $\gcd(F_m,F_n)=F_{\gcd(m,n)}$.

\subsection{Теорема Акра--Бацци (Akra--Bazzi)}
$T(n)=g(n)+\sum a_i T(b_i n+h_i(n))$, $\sum a_i b_i^p=1$ $\Rightarrow$ $T(n)=\Theta(n^p(1+\int_1^n g(u)/u^{p+1}du))$.

\section{Теория графов}

\subsection{Характеризации дерева}
Для $G$ на $n$ вершинах эквивалентны: (1) дерево, (2) связен + $|E|=n-1$, (3) ацикличен + $|E|=n-1$.

\subsection{Теорема Кирхгофа (matrix-tree)}
$\tau(G)=\det\widetilde L_{ii}=\frac1n\lambda_2\cdots\lambda_n$, где $L=D-A$ --- лапласиан. Для $K_n$: $\tau=n^{n-2}$.

\subsection{Эйлеров обход}
Связный граф имеет эйлеров цикл $\iff$ все степени чётны. Путь: ровно 0 или 2 нечётных.

\subsection{Спектры стандартных графов}
$K_n$: $n-1$ (1), $-1$ ($n-1$). $K_{m,n}$: $\pm\sqrt{mn}$, $0$ ($m+n-2$). $C_n$: $2\cos(2\pi k/n)$. $P_n$: $2\cos(\pi k/(n+1))$.

\subsection{Граница Хоффмана для $\alpha(G)$}
$G$ --- $k$-регулярный на $n$ вершинах, $\lambda_n$ --- минимальное собственное значение $\Rightarrow$ $\alpha(G)\le\frac{-n\lambda_n}{k-\lambda_n}$.

\subsection{Граница Уилфа для $\chi(G)$}
$\chi(G)\le 1+\lambda_1(G)$.

\subsection{Условия гамильтоновости (Дирак, Оре)}
Дирак: $\delta\ge n/2$ $\Rightarrow$ гамильтонов. Оре: $\deg u+\deg v\ge n$ для любых несмежных $u,v$ $\Rightarrow$ гамильтонов.

\subsection{Теорема Холла (для графов)}
Двудольный граф $G=(X\cup Y,E)$ имеет паросочетание, насыщающее $X$ $\iff$ $|N(S)|\ge|S|$ для всех $S\subseteq X$.

\section{Комбинаторная геометрия}

\subsection{Теорема Хелли (Helly)}
Выпуклые множества $K_1,\dots,K_m\subseteq\mathbb{R}^d$, $m\ge d+1$. Если любые $d+1$ из них имеют общую точку, то $\bigcap_{i=1}^m K_i\neq\varnothing$.

\textit{Когда использовать.} Задачи о покрытиях, пересечениях выпуклых фигур, расположении точек.

\subsection{Теорема Сильвестра--Галлаи (Sylvester–Gallai)}
Для конечного множества точек на плоскости, не лежащих на одной прямой, существует прямая, содержащая ровно две из них.

\textit{Идея.} Рассмотреть пару (точка, прямая) с минимальным расстоянием; если на прямой $\ge3$ точек, находится другая пара с меньшим расстоянием.

\subsection{Теорема Минковского о выпуклом теле}
$K\subset\mathbb{R}^n$ --- выпуклое центрально-симметричное ($K=-K$) тело с $\operatorname{vol}(K)>2^n$. Тогда $K$ содержит ненулевую целочисленную точку $z\in\mathbb{Z}^n\setminus\{0\}$.

\textit{Когда использовать.} Диофантовы приближения, существование решений систем неравенств, квадратичные формы.

\section{Формула включений-исключений и обращение Мёбиуса}

\subsection{Формула включений-исключений (ФВИ)}
$|\bigcap\overline{A_i}|=\sum_{k=0}^n(-1)^k S_k$, где $S_k=\sum_{|I|=k}|\bigcap_{i\in I}A_i|$. Работает для меры/вероятности.

\subsection{Неравенства Бонферрони}
Чётное усечение $m=2r$: $|\bigcup A_i|\ge\sum_{k=1}^{2r}(-1)^{k+1}S_k$. Нечётное $m=2r+1$: $|\bigcup A_i|\le\sum_{k=1}^{2r+1}(-1)^{k+1}S_k$.

\subsection{Обращение Мёбиуса}
На локально конечном poset: $g(y)=\sum_{x\le y}f(x)$ $\iff$ $f(y)=\sum_{x\le y}\mu(x,y)g(x)$. Булева решётка: $\mu(I,J)=(-1)^{|J\setminus I|}$. Решётка делителей: $\mu(d,n)=\mu(n/d)$.

\subsection{Функция Эйлера $\varphi$}
$\varphi(n)=n\prod_{p\mid n}(1-1/p)=\sum_{d\mid n}\mu(d)n/d$. Обращение: $\sum_{d\mid n}\varphi(d)=n$.

\subsection{Многочлены ладей (rook polynomials)}
$B\subseteq[n]\times[n]$ --- запрет; $r_k(B)$ --- число расстановок $k$ ладей на $B$. $N(B)=\sum_{k=0}^n(-1)^k r_k(B)(n-k)!$ --- число перестановок без запретов.

\subsection{Хроматический многочлен}
$P(G,x)=\sum_{S\subseteq E}(-1)^{|S|}x^{c(V,S)}$, где $c(V,S)$ --- число компонент $(V,S)$. Рекурсия: $P(G,x)=P(G\setminus e,x)-P(G/e,x)$.
