\chapter{Вещественный анализ (одномерный)}

\section{Пределы последовательностей и функций}

\subsection{Теорема Штольца–Чезаро (Stolz–Cesàro)}

\textit{Формулировка.} $b_n$ строго возрастает к $+\infty$ (или $a_n,b_n\to0$, $b_n$ строго убывает). Если $\displaystyle\lim_{n\to\infty}\frac{a_{n+1}-a_n}{b_{n+1}-b_n}=L$, то $\displaystyle\lim_{n\to\infty}\frac{a_n}{b_n}=L$.

\textit{Когда использовать.} Пределы вида $\frac{\infty}{\infty}$ или $\frac{0}{0}$ с разностями. Вычисление $\lim \frac{\ln n}{n}$, $\lim \frac{n}{\sqrt[n]{n!}}$.

\textit{Следствия.} Среднее Чезаро: $a_n\to L\;\Rightarrow\; \frac1n\sum_{k\leq n}a_k\to L$. Геометрическое: $a_{n+1}/a_n\to L\;\Rightarrow\;\sqrt[n]{a_n}\to L$.

\medskip

\section{Теоремы о среднем, Тейлор, правило Лопиталя}

\subsection{Ролль, Лагранж, Коши (MVT)}

\textit{Формулировка.} $f,g\in C[a,b]$, дифференцируемы на $(a,b)$.
\begin{itemize}
  \item \textit{Ролль:} $f(a)=f(b)\;\Rightarrow\;\exists\xi\in(a,b): f'(\xi)=0$.
  \item \textit{Лагранж:} $f(b)-f(a)=f'(\xi)(b-a)$.
  \item \textit{Коши:} $(f(b)-f(a))g'(\xi)=(g(b)-g(a))f'(\xi)$.
\end{itemize}

\textit{Когда использовать.} Оценка приращений, доказательство существования точек с заданным свойством производной, основа для Тейлора и Лопиталя.

\textit{Метод вспомогательной функции.} Чтобы доказать $\exists\xi:\Phi(\xi,f,f',\ldots)=0$, построить $g$ с $g(a)=g(b)$ такую, что $g'=0\Leftrightarrow\Phi=0$, и применить Ролля. Типовые шаблоны: $f'-\lambda f=0$ $\to$ $g=f e^{-\lambda x}$; $f'-f/x=0$ $\to$ $g=f/x$.

\medskip

\subsection{Формула Тейлора (Taylor)}

\textit{Формулировка.} $f(x)=\sum_{k=0}^n\frac{f^{(k)}(a)}{k!}(x-a)^k+R_n(x)$.

Формы остатка:
\begin{itemize}
  \item \textit{Лагранж:} $R_n=\frac{f^{(n+1)}(\xi)}{(n+1)!}(x-a)^{n+1}$.
  \item \textit{Коши:} $R_n=\frac{f^{(n+1)}(\xi)}{n!}(x-\xi)^n(x-a)$.
  \item \textit{Интегральная:} $R_n=\frac1{n!}\int_a^x (x-t)^n f^{(n+1)}(t)\,dt$.
  \item \textit{Пеано:} $R_n=o((x-a)^n)$ при $x\to a$.
\end{itemize}

\textit{Когда использовать.} Оценки через производные, асимптотики, неравенства, экстремумы.

\medskip

\subsection{Правило Лопиталя (L'Hôpital)}

\textit{Формулировка.} $f,g$ дифференцируемы близ $a$, $g'(x)\neq0$. Если $\lim f/g$ вида $\frac00$ или $\frac{*}{\infty}$ и $\lim f'/g'=L$, то $\lim f/g=L$.

\textit{Важно.} Необходимо существование предела $f'/g'$ (обратное неверно). Работает и для $*/\infty$ (без требования $f\to\infty$).

\medskip

\subsection{Теорема Дарбу}

\textit{Формулировка.} $f'$ (даже разрывная) обладает свойством промежуточного значения: если $f$ дифференцируема на $[a,b]$, то $f'$ принимает все значения между $f'(a)$ и $f'(b)$.

\medskip

\section{Неравенства}

\subsection{AM–GM (о средних)}

\textit{Формулировка.} $\frac{x_1+\cdots+x_n}{n}\geq\sqrt[n]{x_1\cdots x_n}$ для $x_i\geq0$. Равенство при $x_1=\cdots=x_n$.

Взвешенное: $\sum w_i x_i\geq\prod x_i^{w_i}$ при $\sum w_i=1$.

\textit{Идея.} Логарифмирование + Йенсен для $\ln$ (вогнут). Или индукция Коши «вперёд-назад».

\medskip

\subsection{Коши–Буняковского–Шварца (Cauchy–Schwarz)}

\textit{Формулировка.} $(\sum a_i b_i)^2\leq(\sum a_i^2)(\sum b_i^2)$. Интегрально: $(\int fg)^2\leq\int f^2\int g^2$. Равенство при $a_i=\lambda b_i$.

\textit{Форма Энгеля (Titu):} $\sum\frac{a_i^2}{b_i}\geq\frac{(\sum a_i)^2}{\sum b_i}$ при $b_i>0$.

\textit{Идея.} Квадратный трёхчлен $\sum(a_i+tb_i)^2\geq0$ имеет неотрицательный дискриминант.

\medskip

\subsection{Йенсен (Jensen)}

\textit{Формулировка.} Для выпуклой $f$: $f(\sum w_i x_i)\leq\sum w_i f(x_i)$ при $\sum w_i=1$. Для вогнутой --- обратный знак. Интегрально: $f(\int x\,d\mu)\leq\int f\,d\mu$.

\textit{Когда использовать.} Любая выпуклая/вогнутая функция в неравенстве.

\textit{Характеризация выпуклости.} $f$ выпукла на $I$ $\iff$ $f''\geq0$ (для $C^2$), либо секущие монотонны, либо $\exists$ опорная прямая в каждой точке. Выпуклая на интервале непрерывна внутри, имеет односторонние производные.

\textit{Эрмит–Адамар:} Для выпуклой $f$ на $[a,b]$: $f\bigl(\frac{a+b}{2}\bigr)\leq\frac1{b-a}\int_a^b f\leq\frac{f(a)+f(b)}{2}$.

\medskip

\subsection{Гёльдер и степенное среднее (Hölder, Power mean)}

\textit{Формулировка (Гёльдер).} При $1/p+1/q=1$: $\sum|a_i b_i|\leq(\sum|a_i|^p)^{1/p}(\sum|b_i|^q)^{1/q}$. Интегрально: $\|fg\|_1\leq\|f\|_p\|g\|_q$.

\textit{Степенное среднее.} $M_r(x;w)=(\sum w_i x_i^r)^{1/r}$ строго возрастает по $r$ (если не все $x_i$ равны). $M_0=\prod x_i^{w_i}$.

\textit{Идея.} Из неравенства Юнга $ab\leq a^p/p+b^q/q$ после нормировки.

\medskip

\subsection{Юнга (Young) и Минковский (Minkowski)}

\textit{Юнг:} $ab\leq\frac{a^p}{p}+\frac{b^q}{q}$ при $1/p+1/q=1$, $a,b\geq0$. $\varepsilon$-версия: $ab\leq\varepsilon a^p/p+\varepsilon^{-q/p}b^q/q$.

\textit{Минковский:} $(\sum(a_i+b_i)^p)^{1/p}\leq(\sum a_i^p)^{1/p}+(\sum b_i^p)^{1/p}$ для $p\geq1$. Интегрально: $\|f+g\|_p\leq\|f\|_p+\|g\|_p$.

\medskip

\subsection{Перестановочное (Rearrangement) и Чебышёв}

\textit{Формулировка.} Для $a_1\leq\cdots\leq a_n$, $b_1\leq\cdots\leq b_n$ и перестановки $\sigma$:
$$\sum a_i b_{n+1-i}\leq\sum a_i b_{\sigma(i)}\leq\sum a_i b_i.$$

\textit{Чебышёв:} $n\sum a_i b_i\geq(\sum a_i)(\sum b_i)$ для одинаково упорядоченных. Интегрально: $(b-a)\int_a^b fg\geq\int_a^b f\int_a^b g$ при одинаковой монотонности.

\textit{Идея.} Тождество $n\sum a_i b_i-\sum a_i\sum b_i=\frac12\sum_{i,j}(a_i-a_j)(b_i-b_j)$.

\medskip

\section{Функциональные уравнения на $\mathbb{R}$}

\subsection{Уравнение Коши (аддитивное)}

\textit{Утверждение.} $f(x+y)=f(x)+f(y)$. При любой регулярности (непрерывность в точке, монотонность, ограниченность на множестве положительной меры): $f(x)=cx$, $c=f(1)$.

\textit{Без регулярности.} Существуют разрывные аддитивные функции (через базис Гамеля). Их график плотен в $\mathbb{R}^2$.

\medskip

\subsection{Уравнения-спутники Коши}

\begin{center}
\begin{tabular}{|c|c|}
\hline
Уравнение & Решение \\
\hline
$f(x+y)=f(x)f(y)$ & $f\equiv0$ или $f(x)=e^{cx}$ \\
$f(xy)=f(x)+f(y)$ на $\mathbb{R}_{>0}$ & $f(x)=c\ln x$ \\
$f(xy)=f(x)f(y)$ на $\mathbb{R}_{>0}$, $f>0$ & $f(x)=x^c$ \\
\hline
\end{tabular}
\end{center}

\textit{Метод.} Сведение к Коши через $\ln$ или $\exp$.

\medskip

\subsection{Уравнение Йенсена}

\textit{Утверждение.} $f\bigl(\frac{x+y}{2}\bigr)=\frac{f(x)+f(y)}{2}$. При непрерывности: $f(x)=ax+b$.

\medskip

\subsection{Уравнение Даламбера (косинусное)}

\textit{Утверждение.} $f(x+y)+f(x-y)=2f(x)f(y)$, $f\not\equiv0$. При непрерывности: $f(x)=\cos(\alpha x)$ или $f(x)=\cosh(\alpha x)$. $f(0)=1$.

\medskip

\subsection{Квадратичное уравнение}

\textit{Утверждение.} $f(x+y)+f(x-y)=2f(x)+2f(y)$. При непрерывности: $f(x)=cx^2$.

\medskip

\subsection{Приёмы: подстановки, сдвиг, сжатие}

\textit{Подстановки.} $x=y=0$ (найти $f(0)$), $y=x$ (связь $f(2x)$ с $f(x)$), $y=-x$ (чётность), симметрия $P(x,y)$ и $P(y,x)$.

\textit{Усиление сдвигом.} Из $f(x+a)\leq f(x)-c$ итерация даёт $f(x+na)\to-\infty$ --- противоречие с $f\geq0$ или ограниченностью.

\textit{Сжатие.} $|f(x)-f(y)|\leq c\cdot q^n$, $q<1$, для всех $n$ $\to$ $f\equiv\text{const}$.

\medskip

\section{Принцип сжимающих отображений (Banach)}

\textit{Формулировка.} $X$ полно, $T:X\to X$, $d(Tx,Ty)\leq k\,d(x,y)$ с $k<1$. Тогда: единственная неподвижная точка $x^*$, итерации $x_{n+1}=Tx_n$ сходятся со скоростью $d(x_n,x^*)\leq\frac{k^n}{1-k}d(x_1,x_0)$.

\subsection{Сжатие в степени}

\textit{Формулировка.} Если $T^n$ --- сжатие для некоторого $n\ge1$, то $T$ имеет единственную неподвижную точку (сам $T$ не обязан быть сжатием).

\medskip

\section{Обыкновенные дифференциальные уравнения}

\subsection{Уравнения с разделяющимися переменными}

$y' = f(x)g(y)$. Решение: $\displaystyle\int\frac{dy}{g(y)} = \int f(x)\,dx$.

\subsection{Линейные уравнения первого порядка}

$y' + p(x)y = q(x)$. Решение: $y = e^{-\int p}\bigl(C + \int q\,e^{\int p}\bigr)$ (метод вариации постоянной или интегрирующий множитель $e^{\int p}$).

\subsection{Уравнение Бернулли}

$y' + p(x)y = q(x)y^n$. Замена $t = y^{1-n}$ сводит к линейному.

\subsection{Уравнение Риккати}

$y' = a(x)y^2 + b(x)y + c(x)$. Частное решение $y_1$ известно $\Rightarrow$ замена $y = y_1 + 1/t$ сводит к линейному.

\subsection{Уравнения в полных дифференциалах}

$M(x,y)\,dx + N(x,y)\,dy = 0$, $\partial_y M = \partial_x N$. Тогда $F(x,y)=C$, где $\partial_x F = M$, $\partial_y F = N$.

Интегрирующий множитель: если $(M_y-N_x)/N$ зависит только от $x$, то $\mu = e^{\int\frac{M_y-N_x}{N}dx}$.

\subsection{Однородные уравнения}

$y' = f(y/x)$. Замена $t = y/x$ сводит к разделяющимся: $t + x t' = f(t)$.

\subsection{Линейные уравнения с постоянными коэффициентами}

$a_n y^{(n)} + \cdots + a_0 y = f(x)$.
\begin{itemize}
\item Однородное: характеристическое уравнение $a_n\lambda^n+\cdots+a_0=0$. Корню $\lambda$ кратности $m$ отвечают решения $x^k e^{\lambda x}$, $k=0,\dots,m-1$.
\item Частное решение: метод неопределённых коэффициентов (для $f = P(x)e^{\alpha x}\cos\beta x$ и т.п.) или вариация постоянных.
\end{itemize}

\subsection{Уравнение Эйлера}

$a_n x^n y^{(n)} + \cdots + a_0 y = f(x)$. Замена $x = e^t$ сводит к уравнению с постоянными коэффициентами.

\subsection{Стандартные замены и приёмы}

Замена $z = \varphi(y)$ упрощает уравнение, если удаётся выразить $y'$ через $z'$:

\begin{itemize}
\item $f'/f = (\ln|f|)'$: если уравнение содержит $\dfrac{y'}{y}$, заменяем $z = \ln|y|$.
\item $ff' = \bigl(\dfrac{f^2}{2}\bigr)'$: если содержит $yy'$, замена $z = y^2$.
\item $f^n f' = \bigl(\dfrac{f^{n+1}}{n+1}\bigr)'$: содержит $y^n y'$, замена $z = y^{n+1}$. Например, $3y^2y' = (y^3)'$ --- коэффициент 3 «появляется» при дифференцировании $y^3$.
\item $f'/f^2 = \bigl(-\dfrac1f\bigr)'$: замена $z = 1/y$ при $y'/y^2$.
\item $f'/f^n = \bigl(\dfrac{1}{(1-n)f^{n-1}}\bigr)'$, $n\neq1$: замена $z = y^{1-n}$.
\item $f'/\sqrt{f} = (2\sqrt{f})'$.
\item $y'/y + p(x) = 0$: интегрирующий множитель $e^{\int p}$ сводит $(y e^{\int p})' = 0$.
\end{itemize}

\section{Гамма-функция, бета-функция, специальные интегралы}

\subsection{Гамма-функция}

$\Gamma(s)=\int_0^\infty t^{s-1}e^{-t}\,dt$ ($\operatorname{Re}s>0$). $\Gamma(s+1)=s\Gamma(s)$. $\Gamma(n+1)=n!$, $\Gamma(1/2)=\sqrt\pi$.

\subsection{Бета-функция}

$B(p,q)=\int_0^1 t^{p-1}(1-t)^{q-1}\,dt=\frac{\Gamma(p)\Gamma(q)}{\Gamma(p+q)}$.

Тригонометрическая форма: $B(p,q)=2\int_0^{\pi/2}\sin^{2p-1}\theta\cos^{2q-1}\theta\,d\theta$.

\subsection{Формула отражения Эйлера}

$\Gamma(x)\Gamma(1-x)=\frac{\pi}{\sin\pi x}$ для $x\notin\mathbb{Z}$.

\subsection{Интеграл Фруллани (Frullani)}

$f$ непрерывна, $A=f(0^+)$, $B=\lim_{x\to\infty}f(x)$. Тогда для $a,b>0$:
$$\int_0^\infty\frac{f(ax)-f(bx)}{x}\,dx=(A-B)\ln\frac{b}{a}.$$

\subsection{Дифференцирование по параметру}

$\frac{d}{dt}\int_{a(t)}^{b(t)} f(x,t)\,dx = f(b(t),t)\,b'(t)-f(a(t),t)\,a'(t)+\int_{a(t)}^{b(t)}\partial_t f(x,t)\,dx$.

\subsection{Интегралы Валлиса (Wallis)}

$W_n=\int_0^{\pi/2}\sin^n x\,dx$. Рекуррентность $W_n=\frac{n-1}{n}W_{n-2}$, асимптотика $W_n\sim\sqrt{\pi/(2n)}$.

\subsection{Симметрия и инверсия}

\textit{Симметрия $x\mapsto a+b-x$:} $\int_a^b f(x)\,dx=\int_a^b f(a+b-x)\,dx$. Если $f(x)+f(a+b-x)=h(x)$, интеграл равен $\frac12\int_a^b h$.

\textit{Инверсия $x\mapsto1/x$:} мера $dx/x$ инвариантна. $\int_0^\infty f(x)/x\,dx=\int_0^\infty f(1/x)/x\,dx$.

\section{Последовательности и ряды}

\subsection{Признак Дирихле}

\textit{Формулировка.} Если частичные суммы $A_n=\sum_{k=1}^n a_k$ ограничены, а $b_n$ монотонно стремится к $0$, то $\sum a_n b_n$ сходится.

\textit{Когда применять.} Знакопеременные ряды (Лейбниц --- частный случай), ряды $\sum (\sin n\theta) b_n$, $\sum (\cos n\theta) b_n$, $\sum z^n b_n$ ($|z|=1$, $z\neq1$) с $b_n\downarrow0$.

\textit{Оценка остатка (Лейбниц).} Для $\sum(-1)^n b_n$ с $b_n\downarrow0$: $|R_n|\le b_{n+1}$. Если $b_n=f(n)$ с $f\in C^1$, то по MVT $b_n-b_{n+1}=f'(\xi_n)$, $\xi_n\in(n,n+1)$ --- для более точных оценок убывания.

\textit{Идея.} Преобразование Абеля: $\sum_{k=1}^N a_k b_k = A_N b_N - \sum_{k=1}^{N-1} A_k (b_{k+1}-b_k)$.

\medskip

\subsection{Признак Абеля}

\textit{Формулировка.} Пусть ряд $\sum a_n$ сходится, а $\{b_n\}$ монотонна и ограничена. Тогда $\sum a_n b_n$ сходится.

\textit{Когда применять.} Домножение сходящегося ряда на колеблющийся множитель; степенные ряды на границе круга сходимости.

\medskip

\subsection{Тест Коши о сгущении}

$a_n\geq0$ монотонно невозрастает $\Rightarrow$ $\sum a_n$ и $\sum 2^k a_{2^k}$ сходятся или расходятся одновременно. Применение: ряды Бертрана $\sum 1/(n\ln^p n)$.

\medskip

\subsection{Тесты Куммера, Раабе, Гаусса}

\textit{Куммер.} $a_n>0$, $p_n>0$, $K_n=p_n\frac{a_n}{a_{n+1}}-p_{n+1}$.
\begin{itemize}
  \item $\liminf K_n>0$ $\Rightarrow$ сходимость.
  \item $\limsup K_n<0$ и $\sum 1/p_n=\infty$ $\Rightarrow$ расходимость.
\end{itemize}

\textit{Раабе.} $R_n=n\bigl(\frac{a_n}{a_{n+1}}-1\bigr)$. Сходимость при $\liminf R_n>1$.

\textit{Гаусс.} $\frac{a_n}{a_{n+1}}=1+\frac{\mu}{n}+O(n^{-1-\delta})$ $\Rightarrow$ сходимость при $\mu>1$.

\medskip

\subsection{Теорема Римана о перестановке}

Условно сходящийся ряд можно переставить так, чтобы его сумма стала любым $L\in\mathbb{R}\cup\{\pm\infty\}$.

\medskip

\subsection{Произведение Коши и теорема Мертенса}

Если $\sum a_n$ сходится абсолютно к $A$, а $\sum b_n$ сходится (условно) к $B$, то произведение Коши $c_n=\sum_{k=0}^n a_k b_{n-k}$ даёт $\sum c_n=AB$.
