\chapter{Real Analysis}
\label{ch:calculus}

\medskip

\textbf{Notation.} $\mathbb{N},\mathbb{Z},\mathbb{Q},\mathbb{R},\mathbb{C}$ --- usual.
$C[a,b]$ --- continuous functions on $[a,b]$; $C^k$ --- $k$ times differentiable with continuous $k$-th derivative.
$f^{(n)}$ --- $n$-th derivative; $f^{(0)}=f$.
$\displaystyle\sum_{n=1}^\infty a_n$ --- series; $\displaystyle\limsup$, $\displaystyle\liminf$.
$\Gamma(s)$ --- Gamma function; $B(p,q)$ --- Beta function.
$O(\cdot),o(\cdot),\sim$ --- asymptotic notation.

\medskip

\section{Limits of sequences and functions}
\label{sec:limits}

\subsection{Stolz--Ces\`aro theorem}
\label{subsec:stolz-cesaro}

\textit{Statement.} $b_n$ strictly increases to $+\infty$ (or $a_n,b_n\to0$, $b_n$ strictly decreases). If $\displaystyle\lim_{n\to\infty}\frac{a_{n+1}-a_n}{b_{n+1}-b_n}=L$, then $\displaystyle\lim_{n\to\infty}\frac{a_n}{b_n}=L$.

\textit{When to use.} Limits of the form $\frac{\infty}{\infty}$ or $\frac{0}{0}$ involving differences. Computing $\lim \frac{\ln n}{n}$, $\lim \frac{n}{\sqrt[n]{n!}}$.

\textit{Corollaries.} Ces\`aro mean: $a_n\to L\;\Rightarrow\; \frac1n\sum_{k\leq n}a_k\to L$. Geometric: $a_{n+1}/a_n\to L\;\Rightarrow\;\sqrt[n]{a_n}\to L$.

\medskip

\section{Mean value theorems, Taylor, L'H\^opital}
\label{sec:mvt-taylor}

\subsection{Rolle, Lagrange, Cauchy (MVT)}
\label{subsec:mvt}

\textit{Statement.} $f,g\in C[a,b]$, differentiable on $(a,b)$.
\begin{itemize}
  \item \textit{Rolle:} $f(a)=f(b)\;\Rightarrow\;\exists\xi\in(a,b): f'(\xi)=0$.
  \item \textit{Lagrange:} $f(b)-f(a)=f'(\xi)(b-a)$.
  \item \textit{Cauchy:} $(f(b)-f(a))g'(\xi)=(g(b)-g(a))f'(\xi)$.
\end{itemize}

\textit{When to use.} Estimating increments, proving existence of points with a given derivative property, foundation for Taylor and L'H\^opital.

\textit{Auxiliary function method.} To prove $\exists\xi:\Phi(\xi,f,f',\ldots)=0$, construct $g$ with $g(a)=g(b)$ such that $g'=0\Leftrightarrow\Phi=0$, then apply Rolle. Typical templates: $f'-\lambda f=0$ $\to$ $g=f e^{-\lambda x}$; $f'-f/x=0$ $\to$ $g=f/x$.

\medskip

\subsection{Taylor's formula}
\label{subsec:taylor}

\textit{Statement.} $f(x)=\sum_{k=0}^n\frac{f^{(k)}(a)}{k!}(x-a)^k+R_n(x)$.

Remainder forms:
\begin{itemize}
  \item \textit{Lagrange:} $R_n=\frac{f^{(n+1)}(\xi)}{(n+1)!}(x-a)^{n+1}$.
  \item \textit{Cauchy:} $R_n=\frac{f^{(n+1)}(\xi)}{n!}(x-\xi)^n(x-a)$.
  \item \textit{Integral:} $R_n=\frac1{n!}\int_a^x (x-t)^n f^{(n+1)}(t)\,dt$.
  \item \textit{Peano:} $R_n=o((x-a)^n)$ as $x\to a$.
\end{itemize}

\textit{When to use.} Estimates via derivatives, asymptotics, inequalities, extrema.

\medskip

\subsection{L'H\^opital's rule}
\label{subsec:lhopital}

\textit{Statement.} $f,g$ differentiable near $a$, $g'(x)\neq0$. If $\lim f/g$ is $\frac00$ or $\frac{*}{\infty}$ and $\lim f'/g'=L$, then $\lim f/g=L$.

\textit{Important.} The limit of $f'/g'$ must exist (the converse does not hold). Also works for $*/\infty$ (without requiring $f\to\infty$).

\medskip

\subsection{Darboux's theorem}
\label{subsec:darboux}

\textit{Statement.} $f'$ (even if discontinuous) has the intermediate value property: if $f$ is differentiable on $[a,b]$, then $f'$ takes every value between $f'(a)$ and $f'(b)$.

\medskip

\section{Inequalities}
\label{sec:inequalities}

\subsection{AM--GM}
\label{subsec:am-gm}

\textit{Statement.} $\frac{x_1+\cdots+x_n}{n}\geq\sqrt[n]{x_1\cdots x_n}$ for $x_i\geq0$. Equality when $x_1=\cdots=x_n$.

Weighted: $\sum w_i x_i\geq\prod x_i^{w_i}$ with $\sum w_i=1$.

\textit{Idea.} Take logs + Jensen for $\ln$ (concave). Or Cauchy's forward--backward induction.

\medskip

\subsection{Cauchy--Schwarz}
\label{subsec:cs-calculus}

\textit{Statement.} $(\sum a_i b_i)^2\leq(\sum a_i^2)(\sum b_i^2)$. Integral form: $(\int fg)^2\leq\int f^2\int g^2$. Equality when $a_i=\lambda b_i$ (see also \ref{subsec:cauchy-schwarz-linalg}).

\textit{Engel form (Titu):} $\sum\frac{a_i^2}{b_i}\geq\frac{(\sum a_i)^2}{\sum b_i}$ for $b_i>0$.

\textit{Idea.} The quadratic polynomial $\sum(a_i+tb_i)^2\geq0$ has non-negative discriminant.

\medskip

\subsection{Jensen}
\label{subsec:jensen}

\textit{Statement.} For convex $f$: $f(\sum w_i x_i)\leq\sum w_i f(x_i)$ with $\sum w_i=1$. For concave --- reverse. Integral form: $f(\int x\,d\mu)\leq\int f\,d\mu$.

\textit{When to use.} Any convex/concave function in an inequality.

\textit{Characterization of convexity.} $f$ is convex on $I$ $\iff$ $f''\geq0$ (if $C^2$), or secants are monotone, or there is a supporting line at each point. A convex function on an interval is continuous in the interior and has one-sided derivatives.

\textit{Hermite--Hadamard:} For convex $f$ on $[a,b]$: $f\bigl(\frac{a+b}{2}\bigr)\leq\frac1{b-a}\int_a^b f\leq\frac{f(a)+f(b)}{2}$.

\medskip

\subsection{H\"older and power mean}
\label{subsec:holder-power-mean}

\textit{Statement (H\"older).} For $1/p+1/q=1$: $\sum|a_i b_i|\leq(\sum|a_i|^p)^{1/p}(\sum|b_i|^q)^{1/q}$. Integral form: $\|fg\|_1\leq\|f\|_p\|g\|_q$.

\textit{Power mean.} $M_r(x;w)=(\sum w_i x_i^r)^{1/r}$ is strictly increasing in $r$ (unless all $x_i$ are equal). $M_0=\prod x_i^{w_i}$.

\textit{Idea.} From Young's inequality $ab\leq a^p/p+b^q/q$ after normalization.

\medskip

\subsection{Young and Minkowski}
\label{subsec:young-minkowski}

\textit{Young:} $ab\leq\frac{a^p}{p}+\frac{b^q}{q}$ for $1/p+1/q=1$, $a,b\geq0$. $\varepsilon$-version: $ab\leq\varepsilon a^p/p+\varepsilon^{-q/p}b^q/q$.

\textit{Minkowski:} $(\sum(a_i+b_i)^p)^{1/p}\leq(\sum a_i^p)^{1/p}+(\sum b_i^p)^{1/p}$ for $p\geq1$. Integral form: $\|f+g\|_p\leq\|f\|_p+\|g\|_p$.

\medskip

\subsection{Rearrangement and Chebyshev}
\label{subsec:rearrangement}

\textit{Statement.} For $a_1\leq\cdots\leq a_n$, $b_1\leq\cdots\leq b_n$ and a permutation $\sigma$:
$$\sum a_i b_{n+1-i}\leq\sum a_i b_{\sigma(i)}\leq\sum a_i b_i.$$

\textit{Chebyshev:} $n\sum a_i b_i\geq(\sum a_i)(\sum b_i)$ for similarly ordered sequences. Integral form: $(b-a)\int_a^b fg\geq\int_a^b f\int_a^b g$ for similarly monotone $f,g$.

\textit{Idea.} Identity $n\sum a_i b_i-\sum a_i\sum b_i=\frac12\sum_{i,j}(a_i-a_j)(b_i-b_j)$.

\medskip

\section{Functional equations on $\mathbb{R}$}
\label{sec:functional-equations}

\subsection{Cauchy's equation (additive)}
\label{subsec:cauchy-additive}

\textit{Claim.} $f(x+y)=f(x)+f(y)$. Under any regularity condition (continuity at a point, monotonicity, boundedness on a set of positive measure): $f(x)=cx$, $c=f(1)$.

\textit{Without regularity.} Discontinuous additive functions exist (via a Hamel basis). Their graph is dense in $\mathbb{R}^2$.

\medskip

\subsection{Related Cauchy-type equations}
\label{subsec:cauchy-related}

\begin{center}
\begin{tabular}{|c|c|}
\hline
Equation & Solution \\
\hline
$f(x+y)=f(x)f(y)$ & $f\equiv0$ or $f(x)=e^{cx}$ \\
$f(xy)=f(x)+f(y)$ on $\mathbb{R}_{>0}$ & $f(x)=c\ln x$ \\
$f(xy)=f(x)f(y)$ on $\mathbb{R}_{>0}$, $f>0$ & $f(x)=x^c$ \\
\hline
\end{tabular}
\end{center}

\textit{Method.} Reduce to Cauchy via $\ln$ or $\exp$.

\medskip

\subsection{Jensen's equation}
\label{subsec:jensen-equation}

\textit{Claim.} $f\bigl(\frac{x+y}{2}\bigr)=\frac{f(x)+f(y)}{2}$. Under continuity: $f(x)=ax+b$.

\medskip

\subsection{D'Alembert's equation (cosine)}
\label{subsec:dalembert}

\textit{Claim.} $f(x+y)+f(x-y)=2f(x)f(y)$, $f\not\equiv0$. Under continuity: $f(x)=\cos(\alpha x)$ or $f(x)=\cosh(\alpha x)$. $f(0)=1$.

\medskip

\subsection{Quadratic equation}
\label{subsec:quadratic-equation}

\textit{Claim.} $f(x+y)+f(x-y)=2f(x)+2f(y)$. Under continuity: $f(x)=cx^2$.

\medskip

\subsection{Techniques: substitutions, shift, contraction}
\label{subsec:functional-tech}

\textit{Substitutions.} $x=y=0$ (find $f(0)$), $y=x$ (relate $f(2x)$ to $f(x)$), $y=-x$ (parity), symmetry $P(x,y)$ and $P(y,x)$.

\textit{Shift amplification.} From $f(x+a)\leq f(x)-c$ iteration gives $f(x+na)\to-\infty$ --- contradiction with $f\geq0$ or boundedness.

\textit{Contraction.} $|f(x)-f(y)|\le c\cdot q^n$ for all $n\ge1$ with $q<1$ $\Rightarrow$ $f(x)=f(y)$ (send $n\to\infty$).

\medskip

\section{Banach fixed-point theorem}
\label{sec:banach}

\textit{Statement.} $X$ is complete, $T:X\to X$, $d(Tx,Ty)\leq k\,d(x,y)$ with $k<1$. Then: unique fixed point $x^*$, iterates $x_{n+1}=Tx_n$ converge with $d(x_n,x^*)\leq\frac{k^n}{1-k}d(x_1,x_0)$.

\subsection{Contraction in power}
\label{subsec:contraction-power}

\textit{Statement.} If $T^n$ is a contraction for some $n\ge1$, then $T$ has a unique fixed point ($T$ itself need not be a contraction).

\medskip

\section{Ordinary differential equations}
\label{sec:ode}

\subsection{Separable equations}
\label{subsec:ode-separable}

\textit{Statement.} $y' = f(x)g(y)$. Solution: $\displaystyle\int\frac{dy}{g(y)} = \int f(x)\,dx$.

\subsection{First-order linear equations}
\label{subsec:ode-linear}

\textit{Statement.} $y' + p(x)y = q(x)$. Solution: $y = e^{-\int p}\bigl(C + \int q\,e^{\int p}\bigr)$ (variation of constants or integrating factor $e^{\int p}$).

\subsection{Bernoulli equation}
\label{subsec:ode-bernoulli}

\textit{Statement.} $y' + p(x)y = q(x)y^n$. Substitution $t = y^{1-n}$ reduces to a linear equation.

\subsection{Riccati equation}
\label{subsec:ode-riccati}

\textit{Statement.} $y' = a(x)y^2 + b(x)y + c(x)$. If a particular solution $y_1$ is known, the substitution $y = y_1 + 1/t$ reduces to a linear equation.

\subsection{Exact equations}
\label{subsec:ode-exact}

\textit{Statement.} $M(x,y)\,dx + N(x,y)\,dy = 0$, $\partial_y M = \partial_x N$. Then $F(x,y)=C$, where $\partial_x F = M$, $\partial_y F = N$.

Integrating factor: if $(M_y-N_x)/N$ depends only on $x$, then $\mu = e^{\int\frac{M_y-N_x}{N}dx}$.

\subsection{Homogeneous equations}
\label{subsec:ode-homogeneous}

\textit{Statement.} $y' = f(y/x)$. Substitution $t = y/x$ reduces to separable: $t + x t' = f(t)$.

\subsection{Linear equations with constant coefficients}
\label{subsec:ode-constant-coeff}

\textit{Statement.} $a_n y^{(n)} + \cdots + a_0 y = f(x)$.
\begin{itemize}
\item Homogeneous: characteristic equation $a_n\lambda^n+\cdots+a_0=0$. A root $\lambda$ of multiplicity $m$ gives solutions $x^k e^{\lambda x}$, $k=0,\dots,m-1$.
\item Particular solution: method of undetermined coefficients (for $f = P(x)e^{\alpha x}\cos\beta x$, etc.) or variation of parameters.
\end{itemize}

\subsection{Euler equation}
\label{subsec:ode-euler}

\textit{Statement.} $a_n x^n y^{(n)} + \cdots + a_0 y = f(x)$. Substitution $x = e^t$ reduces to an equation with constant coefficients.

\subsection{Standard substitutions and tricks}
\label{subsec:ode-substitutions}

The substitution $z = \varphi(y)$ simplifies the equation if $y'$ can be expressed via $z'$:

\begin{itemize}
\item $f'/f = (\ln|f|)'$: if the equation contains $\dfrac{y'}{y}$, set $z = \ln|y|$.
\item $ff' = \bigl(\dfrac{f^2}{2}\bigr)'$: if it contains $yy'$, set $z = y^2$.
\item $f^n f' = \bigl(\dfrac{f^{n+1}}{n+1}\bigr)'$: contains $y^n y'$, set $z = y^{n+1}$.
\item $f'/f^2 = \bigl(-\dfrac1f\bigr)'$: set $z = 1/y$ for $y'/y^2$.
\item $f'/f^n = \bigl(\dfrac{1}{(1-n)f^{n-1}}\bigr)'$, $n\neq1$: set $z = y^{1-n}$.
\item $f'/\sqrt{f} = (2\sqrt{f})'$.
\end{itemize}

\section{Gamma function, beta function, special integrals}
\label{sec:special-functions}

\subsection{Gamma function}
\label{subsec:gamma}

$\Gamma(s)=\int_0^\infty t^{s-1}e^{-t}\,dt$ ($\operatorname{Re}s>0$). $\Gamma(s+1)=s\Gamma(s)$. $\Gamma(n+1)=n!$, $\Gamma(1/2)=\sqrt\pi$.

\subsection{Beta function}
\label{subsec:beta}

$B(p,q)=\int_0^1 t^{p-1}(1-t)^{q-1}\,dt=\frac{\Gamma(p)\Gamma(q)}{\Gamma(p+q)}$.

Trigonometric form: $B(p,q)=2\int_0^{\pi/2}\sin^{2p-1}\theta\cos^{2q-1}\theta\,d\theta$.

\subsection{Euler's reflection formula}
\label{subsec:reflection}

$\Gamma(x)\Gamma(1-x)=\frac{\pi}{\sin\pi x}$ for $x\notin\mathbb{Z}$.

\subsection{Frullani integral}
\label{subsec:frullani}

$f$ is continuous, $A=f(0^+)$, $B=\lim_{x\to\infty}f(x)$. Then for $a,b>0$:
$$\int_0^\infty\frac{f(ax)-f(bx)}{x}\,dx=(A-B)\ln\frac{b}{a}.$$

\subsection{Differentiation under the integral}
\label{subsec:diff-under-integral}

$\frac{d}{dt}\int_{a(t)}^{b(t)} f(x,t)\,dx = f(b(t),t)\,b'(t)-f(a(t),t)\,a'(t)+\int_{a(t)}^{b(t)}\partial_t f(x,t)\,dx$.

\medskip

\subsection{Symmetry $x\mapsto a+b-x$ and inversion $x\mapsto1/x$}
\label{subsec:symmetry-inversion}

\textit{Symmetry $x\mapsto a+b-x$:} $\int_a^b f(x)\,dx=\int_a^b f(a+b-x)\,dx$. If $f(x)+f(a+b-x)=h(x)$, the integral equals $\frac12\int_a^b h$.

\textit{Inversion $x\mapsto1/x$:} the measure $dx/x$ is invariant. $\int_0^\infty f(x)/x\,dx=\int_0^\infty f(1/x)/x\,dx$.

\medskip

\section{Multivariable calculus}
\label{sec:multivariable}

\subsection{Gradient, Jacobian, directional derivative}
\label{subsec:gradient-jacobian}

For $F:\mathbb{R}^n\to\mathbb{R}^m$, $J_F=(\partial_j F_i)$. $D_v F(a)=J_F(a)v$. Chain rule: $J_{F\circ G}=J_F(G)\cdot J_G$.

\subsection{Implicit function theorem}
\label{subsec:implicit-function}

$F:\mathbb{R}^{n+m}\to\mathbb{R}^m$ is $C^1$, $F(a,b)=0$, $\det\partial_y F(a,b)\neq0$. Then near $a$ there exists a unique $C^1$ function $g(x)$ with $F(x,g(x))=0$ and $J_g=-(\partial_y F)^{-1}\partial_x F$.

\textit{When to use.} Showing that an equation defines a $C^1$ curve/surface near a non-degenerate point; computing partial derivatives of implicitly defined functions.

\subsection{Lagrange multipliers}
\label{subsec:lagrange-multipliers}

Extremum of $f(x)$ under $g_1(x)=\cdots=g_k(x)=0$: $\nabla f=\sum\lambda_i\nabla g_i$ at a regular point. The $k$ constraints reduce dimension; $\lambda_i$ are Lagrange multipliers.

\textit{When to use.} Optimization with equality constraints; inequalities by extremizing a function over a sphere, simplex, etc.

\subsection{Change of variables in multiple integrals}
\label{subsec:change-variables}

$x=\Phi(u)$, $\Phi$ is a $C^1$ diffeomorphism:
$$\int_{\Phi(\Omega)} f(x)\,dx = \int_\Omega f(\Phi(u))\,|\det J_\Phi(u)|\,du.$$

\textit{Polar in $\mathbb{R}^2$:} $dx\,dy = r\,dr\,d\theta$. \textit{Spherical in $\mathbb{R}^3$:} $dx\,dy\,dz = r^2\sin\theta\,dr\,d\theta\,d\phi$.

\section{Sequences and series}
\label{sec:series}

\subsection{Dirichlet test}
\label{subsec:dirichlet}

\textit{Statement.} If the partial sums $A_n=\sum_{k=1}^n a_k$ are bounded and $b_n$ decreases monotonically to $0$, then $\sum a_n b_n$ converges.

\textit{When to use.} Alternating series (Leibniz is a special case), series $\sum (\sin n\theta) b_n$, $\sum (\cos n\theta) b_n$, $\sum z^n b_n$ ($|z|=1$, $z\neq1$) with $b_n\downarrow0$.

\textit{Remainder estimate (Leibniz).} For $\sum(-1)^n b_n$ with $b_n\downarrow0$: $|R_n|\le b_{n+1}$. If $b_n=f(n)$ with $f\in C^1$, then by MVT (\ref{subsec:mvt}) $b_n-b_{n+1}=f'(\xi_n)$, $\xi_n\in(n,n+1)$ --- for more precise decay estimates.

\textit{Idea.} Abel transformation: $\sum_{k=1}^N a_k b_k = A_N b_N - \sum_{k=1}^{N-1} A_k (b_{k+1}-b_k)$.

\medskip

\subsection{Abel test}
\label{subsec:abel}

\textit{Statement.} Let $\sum a_n$ converge and $\{b_n\}$ be monotone and bounded. Then $\sum a_n b_n$ converges.

\textit{When to use.} Multiplying a convergent series by an oscillating factor; power series at the boundary of the convergence disk.

\medskip

\subsection{Cauchy condensation test}
\label{subsec:condensation}

$a_n\geq0$ monotonically non-increasing $\Rightarrow$ $\sum a_n$ and $\sum 2^k a_{2^k}$ converge or diverge together. Application: Bertrand series $\sum 1/(n\ln^p n)$.

\medskip

\subsection{Integral test and estimate}
\label{subsec:integral-test}

For $f\ge0$ decreasing on $[1,\infty)$:
$$\int_1^{n+1} f(x)\,dx \le \sum_{k=1}^n f(k) \le f(1)+\int_1^n f(x)\,dx.$$
In the limit: $\sum_{n=1}^\infty f(n)$ converges $\iff$ $\int_1^\infty f(x)\,dx$ converges.

\textit{When to use.} Estimating partial sums of slowly decaying series (e.g.\ $\sum 1/k^s$, $\sum 1/(k\ln k)$). The integral bounds give $O$ and $\Theta$ asymptotics.

\medskip

\subsection{Kummer, Raabe, Gauss tests}
\label{subsec:kummer-raabe-gauss}

\textit{Kummer.} $a_n>0$, $p_n>0$, $K_n=p_n\frac{a_n}{a_{n+1}}-p_{n+1}$.
\begin{itemize}
  \item $\liminf K_n>0$ $\Rightarrow$ convergence.
  \item $\limsup K_n<0$ and $\sum 1/p_n=\infty$ $\Rightarrow$ divergence.
\end{itemize}

\textit{Raabe.} $R_n=n\bigl(\frac{a_n}{a_{n+1}}-1\bigr)$. Convergence when $\liminf R_n>1$.

\textit{Gauss.} $\frac{a_n}{a_{n+1}}=1+\frac{\mu}{n}+O(n^{-1-\delta})$ $\Rightarrow$ convergence when $\mu>1$.

\medskip

\subsection{Riemann rearrangement theorem}
\label{subsec:riemann-rearrangement}

A conditionally convergent series can be rearranged so that its sum becomes any $L\in\mathbb{R}\cup\{\pm\infty\}$.

\medskip

\subsection{Cauchy product and Mertens' theorem}
\label{subsec:mertens}

If $\sum a_n$ converges absolutely to $A$ and $\sum b_n$ converges (conditionally) to $B$, then the Cauchy product $c_n=\sum_{k=0}^n a_k b_{n-k}$ yields $\sum c_n=AB$.
