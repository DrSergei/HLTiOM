\chapter{Теория групп}

\section{Порядки, Лагранж, Коши, циклические группы}

\subsection{Теорема Лагранжа}
$|G|<\infty$, $H\le G$ $\Rightarrow$ $|H|\mid|G|$ и $|G|=[G:H]\cdot|H|$. Следствие: $\operatorname{ord}(g)\mid|G|$, $g^{|G|}=e$.

\subsection{Теорема Коши}
$p$ простое, $p\mid|G|$ $\Rightarrow$ $\exists$ элемент порядка $p$.

\subsection{Формула порядка степени}
$\operatorname{ord}(g)=n<\infty$ $\Rightarrow$ $\operatorname{ord}(g^k)=n/\gcd(k,n)$.

\subsection{Структура подгрупп циклической группы}
$G=\langle g\rangle$, $|G|=n$: любая подгруппа циклическая; для каждого $d\mid n$ ровно одна подгруппа порядка $d$: $\langle g^{n/d}\rangle$.

\subsection{Порядок произведения в абелевой группе}
$G$ абелева, $\operatorname{ord}(a)=m$, $\operatorname{ord}(b)=n$:
\begin{enumerate}
\item $\gcd(m,n)=1$ $\Rightarrow$ $\operatorname{ord}(ab)=mn$.
\item В $\langle a,b\rangle$ $\exists$ элемент порядка $\operatorname{lcm}(m,n)$.
\end{enumerate}
Осторожно: в неабелевой группе неверно ($S_3$).

\subsection{Критерий цикличности}
Конечная $G$ циклическая $\Leftrightarrow$ для каждого $d\mid|G|$ уравнение $x^d=e$ имеет $\le d$ решений. Равносильно: для каждого $d$ не более одной подгруппы порядка $d$.

\subsection{Теорема Фробениуса}
$G$ конечна, $n\mid|G|$ $\Rightarrow$ число решений $x^n=e$ делится на $n$.

\subsection{Группа порядка $p^2$ абелева}
$|G|=p^2$ $\Rightarrow$ $G$ абелева. С точностью до изоморфизма: $\mathbb Z/p^2$ и $\mathbb Z/p\times\mathbb Z/p$.

\subsection{Группы порядка $pq$}
$|G|=pq$, $p<q$ простые. Тогда:
\begin{enumerate}
\item $p\nmid q-1$ $\Rightarrow$ $G\cong\mathbb Z/(pq)$.
\item $p\mid q-1$ $\Rightarrow$ две группы: циклическая и $\mathbb Z/q\rtimes\mathbb Z/p$ (неабелева).
\end{enumerate}

\subsection{Подгруппа наименьшего простого индекса нормальна}
$[G:H]=p$ --- наименьший простой делитель $|G|$ $\Rightarrow$ $H\trianglelefteq G$.

\subsection{Формула $|HK|$}
$H,K\le G$ конечны $\Rightarrow$ $|HK|=|H|\cdot|K|/|H\cap K|$.

\subsection{Индекс пересечения}
$[G:H\cap K]\le[G:H]\cdot[G:K]$.

\subsection{Китайская теорема для циклических групп}
$\gcd(m,n)=1$ $\Rightarrow$ $\mathbb Z/m\times\mathbb Z/n\cong\mathbb Z/(mn)$.

\subsection{$g^2=e$ всюду $\Rightarrow$ $G$ абелева}
Если $\forall g\in G\;g^2=e$, то $G$ абелева. Если $G$ конечна --- $G\cong(\mathbb Z/2)^k$.

\section{Действие группы: орбита-стабилизатор, уравнение классов}

\subsection{Теорема об орбите и стабилизаторе}
$G$ действует на $X$, $x\in X$. Отображение $gG_x\mapsto gx$ --- биекция $G/G_x\to Gx$. При $|G|<\infty$:
$$|Gx|=[G:G_x]=\frac{|G|}{|G_x|}.$$

\subsection{Уравнение классов (сопряжение)}
$G$ действует на себе сопряжением $g\cdot h=ghg^{-1}$.
\begin{enumerate}
\item $|[h]|=[G:C_G(h)]$.
\item $|G|=|Z(G)|+\sum_i[G:C_G(g_i)]$ (по нецентральным классам).
\item Число сопряжённых подгруппы $H$ равно $[G:N_G(H)]$.
\end{enumerate}

\subsection{Лемма о неподвижных точках $p$-группы}
$P$ --- $p$-группа, действует на конечном $X$ $\Rightarrow$ $|X^P|\equiv|X|\pmod p$. Если $p\nmid|X|$, то есть неподвижная точка.

\subsection{Центр конечной нетривиальной $p$-группы нетривиален}
$|G|=p^n$, $n\ge1$ $\Rightarrow$ $p\mid|Z(G)|$ (в частности $Z(G)\neq\{e\}$).

\subsection{Лемма Бёрнсайда (Коши--Фробениуса)}
$$|X/G|=\frac1{|G|}\sum_{g\in G}|X^g|,$$
где $X^g$ --- неподвижные точки $g$. Вариант Пойа: $G\le S_n$, число орбит на функциях $[n]\to Y$, $|Y|=k$: $\frac1{|G|}\sum_g k^{c(g)}$, $c(g)$ --- число циклов $g$.

\subsection{Теорема Кэли (обобщённая)}
\begin{enumerate}
\item $G$ вкладывается в $S_{|G|}$ (левым сдвигом).
\item Если $[G:H]=n$, действие на $G/H$ даёт $\varphi:G\to S_n$ с $\ker\varphi=\bigcap_{g}gHg^{-1}$. Ядро --- нормальная подгруппа в $H$, $[G:\ker\varphi]\mid n!$.
\end{enumerate}

\section{Классы сопряжённости, нормализатор, централизатор}

\subsection{Размер класса делит индекс центра}
$|\operatorname{cl}(g)|\mid[G:Z(G)]$. $|\operatorname{cl}(g)|=1 \iff g\in Z(G)$.

\subsection{$G/Z(G)$ циклична $\Rightarrow$ $G$ абелева}
Если $G/Z(G)$ циклическая, то $G$ абелева (и $G/Z(G)$ тривиальна).

\subsection{Теорема N/C}
$H\le G$ $\Rightarrow$ $N_G(H)/C_G(H)\hookrightarrow\operatorname{Aut}(H)$. Частный случай: $G/Z(G)\cong\operatorname{Inn}(G)\le\operatorname{Aut}(G)$.

\subsection{Нормализатор циклической подгруппы}
$\operatorname{ord}(g)=n$ $\Rightarrow$ $N_G(\langle g\rangle)/C_G(g)\hookrightarrow(\mathbb Z/n)^\times$. Следствие: $[N_G(\langle g\rangle):C_G(g)]\mid\varphi(n)$.

\subsection{Размер класса в $p$-группе}
$|G|=p^n$, $g\notin Z(G)$ $\Rightarrow$ $|\operatorname{cl}(g)|=p^k$, $k\ge1$.

\section{$S_n$ и $A_n$: знак, структура, сопряжённость}

\subsection{Знак как гомоморфизм}
Единственный $\operatorname{sgn}:S_n\to\{\pm1\}$ c $\operatorname{sgn}(\text{транспозиция})=-1$. $\ker\operatorname{sgn}=A_n$, $[S_n:A_n]=2$.

\subsection{Цикловой тип --- полный инвариант сопряжённости в $S_n$}
$\sigma,\tau\in S_n$ сопряжены $\Leftrightarrow$ одинаковый цикловой тип. Размер класса с типом $1^{m_1}2^{m_2}\cdots n^{m_n}$: $\dfrac{n!}{\prod_k m_k!\,k^{m_k}}$.
$|C_{S_n}(\sigma)|=\prod_k m_k!\,k^{m_k}$.

\subsection{Формула сопряжения}
$\rho\,(a_1\dots a_k)\,\rho^{-1}=(\rho(a_1)\dots\rho(a_k))$.

\subsection{Знак через цикловую структуру}
$\operatorname{sgn}(\sigma)=(-1)^{n-c(\sigma)}=(-1)^{\sum(\lambda_i-1)}$, где $c(\sigma)$ --- число циклов. $k$-цикл имеет знак $(-1)^{k-1}$.

\subsection{Генераторы $S_n$}
$S_n$ порождается: всеми транспозициями; соседними транспозициями; парой $(1\,2)$ и $(1\,2\,\dots\,n)$.

\subsection{Генераторы $A_n$}
$A_n$ при $n\ge3$ порождается тройными циклами $(1\,2\,k)$, $k=3,\dots,n$.

\subsection{Расщепление классов $S_n\to A_n$}
Класс $\sigma\in A_n$ в $S_n$ распадается на два класса в $A_n$ $\Leftrightarrow$ цикловой тип $\sigma$ состоит из попарно различных нечётных длин.

\subsection{Простота $A_n$ ($n\ge5$)}
$A_n$ проста при $n\ge5$.

\subsection{Нормальные подгруппы $S_n$}
$n=1,2$: $\{e\},S_n$. $n=3$: $\{e\},A_3,S_3$. $n=4$: $\{e\},V_4,A_4,S_4$. $n\ge5$: $\{e\},A_n,S_n$.

\subsection{Коммутант}
$[S_n,S_n]=A_n$ ($n\ge2$). $[A_n,A_n]=A_n$ при $n\ge5$ (совершенна). $[A_4,A_4]=V_4$.

\subsection{Разрешимость $S_n$, $A_n$}
$S_n$ и $A_n$ разрешимы $\Leftrightarrow$ $n\le4$.

\section{Теоремы Силова и $p$-группы}

\subsection{Теоремы Силова}
$|G|=p^a m$, $\gcd(p,m)=1$.
\begin{itemize}
\item \textit{I (существование).} Для каждого $0\le k\le a$ есть подгруппа порядка $p^k$. Подгруппа порядка $p^a$ --- силовская $p$-подгруппа.
\item \textit{II (сопряжённость).} Все силовские $p$-подгруппы сопряжены. Любая $p$-подгруппа лежит в некоторой силовской.
\item \textit{III (счёт).} $n_p\equiv1\pmod p$, $n_p\mid m$, $n_p=[G:N_G(P)]$.
\end{itemize}

\subsection{Аргумент Фраттини}
$N\trianglelefteq G$, $P\in\operatorname{Syl}_p(N)$ $\Rightarrow$ $G=N\cdot N_G(P)$.

\subsection{$p$-группа нильпотентна}
$|G|=p^n$ $\Rightarrow$ верхний центральный ряд достигает $G$ за $\le n$ шагов. Следствие: в $p$-группе любая собственная $H$ строго меньше $N_G(H)$.

\subsection{Максимальная подгруппа $p$-группы}
$|G|=p^n$, $M$ максимальна $\Rightarrow$ $M\trianglelefteq G$ и $[G:M]=p$.

\subsection{Подгруппа Фраттини}
$\Phi(G)=\bigcap M$ (пересечение всех максимальных). Для $p$-группы: $\Phi(G)=[G,G]\cdot G^p$.
$G/\Phi(G)\cong(\mathbb F_p)^d$, $d$ --- минимальное число образующих.

\subsection{Цепочка нормальных подгрупп в $p$-группе}
$|G|=p^n$ $\Rightarrow$ $\exists$ $\{e\}=N_0\lneq\cdots\lneq N_n=G$ с $|N_k|=p^k$ и $N_k\trianglelefteq G$.

\section{Конечные абелевы группы}

\subsection{Теорема о структуре}
Любая конечная абелева $G$:
\begin{itemize}
\item \textit{Инвариантные множители:} $G\cong\mathbb Z/d_1\oplus\cdots\oplus\mathbb Z/d_k$, $1<d_1\mid d_2\mid\cdots\mid d_k$.
\item \textit{Элементарные делители:} $G\cong\bigoplus\mathbb Z/p_i^{a_i}$ --- однозначно с точностью до перестановки.
\end{itemize}

\subsection{Разложение на $p$-силовские}
$G$ абелева $\Rightarrow$ $G=\bigoplus_{p\mid|G|}G_p$, где $G_p$ --- $p$-силовская.

\subsection{$G[n]$ в абелевой группе}
$G\cong\bigoplus\mathbb Z/p_i^{a_i}$ $\Rightarrow$ $|G[n]|=\prod_i\gcd(n,p_i^{a_i})$.

\subsection{Обращение Лагранжа в абелевом случае}
$G$ абелева, $d\mid|G|$ $\Rightarrow$ $\exists$ $H\le G$ порядка $d$.

\subsection{Гауссов бином}
Число подгрупп порядка $p^k$ в $(\mathbb Z/p)^n$ равно $\binom{n}{k}_p=\prod_{i=0}^{k-1}\frac{p^{n-i}-1}{p^{k-i}-1}$.

\subsection{Базисная теорема Бернсайда}
$G$ --- абелева $p$-группа. $d(G)=\dim_{\mathbb F_p}G/pG$ --- минимальное число образующих.

\subsection{Двойственность характеров}
$G$ абелева. $\widehat G=\operatorname{Hom}(G,\mathbb C^\times)\cong G$.
Ортогональность: $\sum_g\chi(g)\overline{\psi(g)}=|G|\cdot[\chi=\psi]$; $\sum_\chi\chi(g)\overline{\chi(h)}=|G|\cdot[g=h]$.

\subsection{Нормальная форма Смита}
$M\in\mathbb Z^{m\times n}$ приводится к $\operatorname{diag}(d_1,\dots,d_r,0,\dots)$ с $d_1\mid d_2\mid\cdots\mid d_r$ умножением на $GL_m(\mathbb Z),GL_n(\mathbb Z)$. Применение: если $G\cong\mathbb Z^n/M\mathbb Z^m$, то $G\cong\mathbb Z^{n-r}\oplus\bigoplus\mathbb Z/d_i$.

\section{Основные методы и приёмы}

\begin{itemize}
\item \textit{Подсчёт Силовских.} $n_p\equiv1\pmod p$, $n_p\mid m$ часто фиксирует $n_p$. $n_p=1$ $\Rightarrow$ силовская нормальна $\Rightarrow$ группа не проста.
\item \textit{Подсчёт элементов.} Если $n_p$ велико, элементы порядка $p$ занимают много места, оставляя мало для других --- часто $n_q=1$.
\item \textit{Уравнение классов.} $|G|=|Z(G)|+\sum[G:C_G(g)]$ --- основа $p$-групповой индукции.
\item \textit{Двойной счёт.} Ядро леммы Бёрнсайда; подсчёт пар $(g,x)$ с $gx=x$.
\item \textit{Действие на смежных классах.} Переводит подгруппу $H$ в гомоморфизм $G\to S_{[G:H]}$. Ядро --- $\operatorname{core}_G(H)$, нормальная подгруппа внутри $H$.
\item \textit{Теорема N/C.} $N_G(H)/C_G(H)\hookrightarrow\operatorname{Aut}(H)$. Ограничивает нормализатор через автоморфизмы.
\item \textit{Сведение к $p$-частям.} В абелевом случае задача разбивается на независимые $p$-компоненты.
\end{itemize}
