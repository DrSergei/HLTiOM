\chapter{Number Theory}
\label{ch:numbers}

\medskip

\textbf{Notation.} $\mathbb{N},\mathbb{Z},\mathbb{Q},\mathbb{R},\mathbb{C}$ --- usual.
$v_p(n)$ --- exponent of $p$ in $n$ (valuation).
$a\mid b$ --- $a$ divides $b$; $p^k\parallel n$ means $p^k\mid n$, $p^{k+1}\nmid n$.
$\operatorname{ord}_n(a)$ --- multiplicative order of $a$ modulo $n$.
$\varphi(n)$ --- Euler totient; $\mu(n)$ --- M\"obius function.
$\left(\frac{a}{p}\right)$ --- Legendre symbol; $\left(\frac{a}{n}\right)$ --- Jacobi symbol.
$\mathbb{Z}/n\mathbb{Z}$ (or $\mathbb{Z}_n$) --- integers modulo $n$.

\medskip

\section{LTE (Lifting The Exponent)}

\subsection{LTE for odd $p$, difference}
\label{subsec:lte-odd-diff}

\textit{Statement.} Let $p$ be an odd prime, $x,y\in\mathbb{Z}$, $n\ge 1$, $p\mid x-y$, $p\nmid xy$. Then
$$v_p(x^n-y^n)=v_p(x-y)+v_p(n).$$

\textit{When to use.} Problems like ``prove $p^k\mid a^n-b^n$'', ``find all $n$ with $p^k\mid a^n\pm b^n$''.

\subsection{LTE for odd $p$, sum}
\label{subsec:lte-odd-sum}

\textit{Statement.} $p$ odd, $n$ odd, $p\mid x+y$, $p\nmid xy$. Then
$$v_p(x^n+y^n)=v_p(x+y)+v_p(n).$$

\subsection{LTE for $p=2$}
\label{subsec:lte-p2}

\textit{Statement.} $x,y$ odd, $n\ge 1$.
\begin{itemize}
\item $n$ odd: $v_2(x^n-y^n)=v_2(x-y)$.
\item $n$ even: $v_2(x^n-y^n)=v_2(x-y)+v_2(x+y)+v_2(n)-1$.
\item If $4\mid x-y$, then $v_2(x^n-y^n)=v_2(x-y)+v_2(n)$.
\end{itemize}
For sum: when $n$ is odd $v_2(x^n+y^n)=v_2(x+y)$; when $n$ is even $v_2(x^n+y^n)=1$.

\subsection{Legendre's formula}
\label{subsec:legendre}

$$v_p(n!)=\sum_{k\ge1}\left\lfloor\frac{n}{p^k}\right\rfloor=\frac{n-s_p(n)}{p-1},$$
where $s_p(n)$ is the sum of digits of $n$ in base $p$.

\subsection{Kummer's theorem}
\label{subsec:kummer}

$v_p\binom{m+n}{n}$ equals the number of carries when adding $m$ and $n$ in base $p$. Equivalently:
$$v_p\binom{m+n}{n}=\frac{s_p(m)+s_p(n)-s_p(m+n)}{p-1}.$$

\subsection{Lifting the order modulo $p^k$}
\label{subsec:lift-order}

Let $p$ be odd, $p\nmid a$, $d=\operatorname{ord}_p(a)$, $c=v_p(a^d-1)\ge1$. For all $k\ge1$:
$$\operatorname{ord}_{p^k}(a)=d\cdot p^{\max(0,\,k-c)}.$$
For $p=2$: when $a\equiv1\pmod4$ we have $\operatorname{ord}_{2^k}(a)=2^{\max(0,\,k-v_2(a-1))}$.

\subsection{Zsigmondy's theorem}
\label{subsec:zsigmondy}

$a>b\ge1$, $\gcd(a,b)=1$, $n\ge2$. The number $a^n-b^n$ has a prime divisor that does not divide $a^k-b^k$ for any $k<n$ (primitive). Exceptions:
\begin{itemize}
\item $n=2$ and $a+b$ is a power of $2$;
\item $(a,b,n)=(2,1,6)$.
\end{itemize}

Analog for sum: for $n\ge2$, $a>b\ge1$, $\gcd(a,b)=1$, $a^n+b^n$ has a primitive prime divisor, except for $n=2$, $a+b$ a power of $2$, or $(a,b,n)=(2,1,3)$.

\section{Diophantine tricks: descent, moduli, estimates}

\subsection{Infinite descent}
\label{subsec:descent}

\textit{Idea.} If from any integer solution one can construct a solution with strictly smaller positive measure, then no solutions exist.

\textit{When to use.} Equations like $x^4+y^4=z^2$, $x^2+y^2=3z^2$, ``there is no $n>1$ with $n\mid2^n-1$''.

\subsection{Modulo obstruction}
\label{subsec:mod-obstruction}

If the equation $F=0$ has no solutions modulo $n$, then it has no solutions in $\mathbb{Z}$ either.

\textit{Useful tables.}
$$x^2\bmod3\in\{0,1\},\; x^2\bmod4\in\{0,1\},\; x^2\bmod8\in\{0,1,4\}.$$
Consequences: $a^2+b^2\not\equiv3\pmod4$, sum of three squares is not $\equiv7\pmod8$.

\subsection{$p$-adic balance}
\label{subsec:padic-balance}

If $A=B$ in $\mathbb{Z}$, then $v_p(A)=v_p(B)$ for every $p$. Usually combined with Legendre's formula (\ref{subsec:legendre}) and LTE (\ref{subsec:lte-odd-diff}).

\subsection{Hensel's lemma}
\label{subsec:hensel}

Let $f\in\mathbb{Z}[x]$, $p$ prime. If $f(a)\equiv0\pmod p$ and $f'(a)\not\equiv0\pmod p$, then for each $k\ge1$ there exists a unique $a_k\in\mathbb{Z}/p^k$ with $a_k\equiv a\pmod p$ and $f(a_k)\equiv0\pmod{p^k}$.

\subsection{Chinese Remainder Theorem (CRT)}
\label{subsec:crt}

$n_1,\dots,n_k$ are pairwise coprime. For any $a_i$ there exists $x$ with $x\equiv a_i\pmod{n_i}$, unique modulo $\prod n_i$.

\subsection{Vieta jumping}
\label{subsec:vieta-jump}

The equation is symmetric in $a,b$ and quadratic in $a$ for fixed $b$. If $(a,b)$ is a solution, then the second root $a'$ is also an integer solution. Usually $|a'|<|a|$, giving a descent.

\subsection{Sophie Germain identity}
\label{subsec:sophie-germain}

$$a^4+4b^4=(a^2+2ab+2b^2)(a^2-2ab+2b^2).$$

\textit{When to use.} Factorization of $n^4+4$ and $n^4+4^n$ (for odd $n$).

\subsection{Order obstruction}
\label{subsec:order-obstruction}

If $\gcd(a,p)=1$ and $d=\operatorname{ord}_p(a)$, then $a^n\equiv1\pmod p\iff d\mid n$. By Fermat's little theorem $d\mid p-1$.

\subsection{Pell equations}
\label{subsec:pell}

$D>0$ is not a square. All positive solutions of $x^2-Dy^2=1$:
$$x_n+y_n\sqrt{D}=(x_1+y_1\sqrt{D})^n,$$
where $(x_1,y_1)$ is the minimal positive solution.

\section{Quadratic residues}
\label{sec:quadratic-residues}

\subsection{Euler's criterion}
\label{subsec:euler-criterion}

For odd prime $p$ and $p\nmid a$:
$$a^{(p-1)/2}\equiv\left(\frac{a}{p}\right)\pmod p.$$
Corollary: $\bigl(\frac{-1}{p}\bigr)=(-1)^{(p-1)/2}$.

\subsection{Quadratic reciprocity law}
\label{subsec:quad-reciprocity}

For distinct odd primes $p,q$:
$$\left(\frac{p}{q}\right)\left(\frac{q}{p}\right)=(-1)^{\frac{p-1}{2}\cdot\frac{q-1}{2}}.$$

\textit{Supplementary.}
$$\left(\frac{-1}{p}\right)=\begin{cases}+1,&p\equiv1\pmod4,\\-1,&p\equiv3\pmod4.\end{cases}$$
$$\left(\frac{2}{p}\right)=\begin{cases}+1,&p\equiv\pm1\pmod8,\\-1,&p\equiv\pm3\pmod8.\end{cases}$$

\subsection{Jacobi symbol}
\label{subsec:jacobi}

For odd $n=\prod p_i^{a_i}$: $\bigl(\frac{a}{n}\bigr)=\prod\bigl(\frac{a}{p_i}\bigr)^{a_i}$. It is multiplicative in both arguments; the same reciprocity law holds. Important: $\bigl(\frac{a}{n}\bigr)=1$ does not imply $a$ is a square modulo $n$.

\subsection{Structure of squares in $(\mathbb{Z}/p^k)^\times$}
\label{subsec:squares-mod-pk}

$p$ odd, $\gcd(a,p)=1$: $a$ is a square in $(\mathbb{Z}/p^k)^\times$ $\iff$ $\bigl(\frac{a}{p}\bigr)=+1$.
For $p=2$ ($k\ge3$): odd $a$ is a square modulo $2^k$ $\iff$ $a\equiv1\pmod8$.

\subsection{Sum of Legendre symbols}
\label{subsec:sum-legendre}

With $p\nmid a$: $\sum_{x=0}^{p-1}\bigl(\frac{ax+b}{p}\bigr)=0$.
For $f(x)=ax^2+bx+c$, $p\nmid a$, $D=b^2-4ac$:
$$\sum_{x=0}^{p-1}\left(\frac{f(x)}{p}\right)=\begin{cases}
-\bigl(\frac{a}{p}\bigr), & p\nmid D,\\
(p-1)\bigl(\frac{a}{p}\bigr), & p\mid D.
\end{cases}$$

\section{Order of an element, primitive roots}
\label{sec:primitive-roots}

\subsection{Properties of order}
\label{subsec:order-properties}

$\gcd(a,n)=1$, $d=\operatorname{ord}_n(a)$.
\begin{enumerate}
\item $a^k\equiv1\pmod n \iff d\mid k$.
\item $d\mid\varphi(n)$.
\item $\operatorname{ord}_n(a^k)=d/\gcd(d,k)$.
\item If $\gcd(\operatorname{ord}_n a,\operatorname{ord}_n b)=1$, then $\operatorname{ord}_n(ab)=\operatorname{ord}_n(a)\cdot\operatorname{ord}_n(b)$.
\item In an abelian group there exists an element of order $\operatorname{lcm}(\operatorname{ord}_n a,\operatorname{ord}_n b)$.
\end{enumerate}

\subsection{Existence of primitive roots}
\label{subsec:primitive-existence}

For any prime $p$, the group $(\mathbb{Z}/p)^\times$ is cyclic. The number of primitive roots mod $p$ is $\varphi(p-1)$.
\textit{Test.} $g$ is a primitive root $\iff$ $g^{(p-1)/q}\not\equiv1\pmod p$ for every prime $q\mid p-1$.

\subsection{Structure of $(\mathbb{Z}/n)^\times$}
\label{subsec:zn-structure}

$(\mathbb{Z}/n)^\times$ is cyclic $\iff$ $n\in\{1,2,4,p^k,2p^k\}$ ($p$ odd prime).
For odd $p$: $(\mathbb{Z}/p^k)^\times\cong\mathbb{Z}/p^{k-1}(p-1)$.
For $p=2$, $k\ge3$: $(\mathbb{Z}/2^k)^\times\cong\mathbb{Z}/2\times\mathbb{Z}/2^{k-2}$.

\subsection{Smallest prime divisor trick}
\label{subsec:smallest-prime}

$n\ge2$, $p$ the smallest prime divisor of $n$, $\gcd(a,n)=1$.
\begin{itemize}
\item If $n\mid a^n-1$, then $a\equiv1\pmod p$.
\item If $n$ is odd and $n\mid a^n+1$, then $a\equiv-1\pmod p$.
\end{itemize}
\textit{When to use.} IMO 1990{-}3 ($n^2\mid2^n+1$), problem ``there is no $n>1$ with $n\mid2^n-1$''.

\subsection{Carmichael function}
\label{subsec:carmichael}

$$\lambda(p^k)=\begin{cases}p^{k-1}(p-1), & p\text{ odd or }p^k\in\{2,4\},\\ 2^{k-2}, & p=2,\ k\ge3,\end{cases}$$
$$\lambda(n)=\operatorname{lcm}\bigl(\lambda(p_1^{a_1}),\dots,\lambda(p_r^{a_r})\bigr).$$
In $(\mathbb{Z}/n)^\times$ there always exists an element of order $\lambda(n)$.

\subsection{Prime divisors of cyclotomic polynomials}
\label{subsec:cyclotomic-prime}

$p\nmid a$. If $p\mid\Phi_n(a)$, then either $\operatorname{ord}_p(a)=n$ ($\implies p\equiv1\pmod n$), or $p\mid n$.

\subsection{Infinitely many primes $\equiv1\pmod n$}
\label{subsec:inf-primes-1modn}

For any $n\ge1$ there are infinitely many primes $p\equiv1\pmod n$. Idea: take a prime divisor of $\Phi_n(M)$ (see \ref{subsec:cyclotomic-prime}).

\subsection{Index and $k$-th power criterion}
\label{subsec:kth-power}

$p$ odd, $g$ a primitive root mod $p$. The map $\operatorname{ind}_g:(\mathbb{Z}/p)^\times\to\mathbb{Z}/(p-1)$ is an isomorphism.
$a$ is a $k$-th power $\iff$ $a^{(p-1)/\gcd(k,p-1)}\equiv1\pmod p$.
The number of $k$-th powers in $(\mathbb{Z}/p)^\times$ is $(p-1)/\gcd(k,p-1)$.

\subsection{Power sums modulo a prime}
\label{subsec:power-sums-mod}

For a prime $p$ and $m\ge0$:
$$\sum_{a=1}^{p-1}a^m\equiv\begin{cases}
-1\pmod p, & m\ge1\text{ and }(p-1)\mid m,\\
0\pmod p, & \text{otherwise}.
\end{cases}$$

\subsection{Generalized Wilson theorem}
\label{subsec:generalized-wilson}

$$\prod_{\substack{1\le a\le n\\\gcd(a,n)=1}}a\equiv\begin{cases}
-1\pmod n, & n\in\{2,4,p^k,2p^k\},\\
+1\pmod n, & \text{otherwise}.
\end{cases}$$
