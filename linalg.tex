\chapter{Линейная алгебра}

\section{Характеристический и минимальный многочлен. Кэли--Гамильтон}

\subsection{Теорема Кэли--Гамильтона}
\begin{description}
  \item[Формулировка:] Для любой квадратной матрицы $A$ над коммутативным кольцом: $\chi_A(A)=0$, где $\chi_A(t)=\det(tI-A)$.
  \item[Когда использовать:] Сведение степеней $A^k$, $k\ge n$ к линейным комбинациям $I,A,\dots,A^{n-1}$. Выражение $A^{-1}$ как многочлена от $A$ степени $\le n-1$.
  \item[Идея:] $(tI-A)\cdot\operatorname{adj}(tI-A)=\chi_A(t)I$. Подстановка $t\mapsto A$ корректна, так как присоединённая матрица --- многочлен от $A$.
\end{description}

\subsection{Минимальный многочлен $\mu_A$}
\begin{description}
  \item[Формулировка:] Нормированный многочлен наименьшей степени с $\mu_A(A)=0$. Делит $\chi_A$, имеет те же корни (с возможно меньшими кратностями). Кратность $\lambda$ в $\mu_A$ равна размеру крупнейшей жордановой клетки для $\lambda$.
  \item[Когда использовать:] Критерий диагонализуемости, анализ спектра.
\end{description}

\subsection{Критерий диагонализуемости}
\begin{description}
  \item[Формулировка:] $A$ диагонализуема над $F$ $\iff$ $\mu_A$ раскладывается над $F$ на \textbf{различные} линейные множители.
  \item[Когда использовать:] Проверка, что $A$ можно диагонализовать (идемпотенты, инволюции, периодические матрицы).
  \item[Идея:] Если $\mu_A=\prod(t-\lambda_i)$, $\lambda_i$ различны, то китайская теорема об остатках даёт $F[t]/(\mu_A)\cong\prod F[t]/(t-\lambda_i)$~--- разложение в сумму собственных подпространств.
\end{description}

\section{След (trace) и спектр произведения}

\subsection{Свойства следа}
\begin{description}
  \item[Формулировка:] $\operatorname{tr}(AB)=\operatorname{tr}(BA)$, $\operatorname{tr}(ABC)=\operatorname{tr}(BCA)=\operatorname{tr}(CAB)$. $\operatorname{tr}(A)=\sum\lambda_i$, $\operatorname{tr}(A^k)=\sum\lambda_i^k$ (в алгебраическом замыкании). След инвариантен под подобием.
  \item[Когда использовать:] Циклическая перестановка в следе, связь следа с собственными значениями.
\end{description}

\subsection{Спектр $AB$ vs $BA$}
\begin{description}
  \item[Формулировка:] $A\in M_{m\times n}$, $B\in M_{n\times m}$: $\det(\lambda I_n-BA)=\lambda^{n-m}\det(\lambda I_m-AB)$. Ненулевые собственные значения $AB$ и $BA$ совпадают с кратностями. При $m=n$: $\chi_{AB}=\chi_{BA}$.
  \item[Когда использовать:] Сведение задачи к матрице меньшего размера. Тождество $\det(I+AB)=\det(I+BA)$.
  \item[Идея:] Для обратимой $A$: $BA=A^{-1}(AB)A$~--- подобие. Общий случай~--- аргумент плотности.
\end{description}

\subsection{Тождества Ньютона}
\begin{description}
  \item[Формулировка:] $p_k=\operatorname{tr}(A^k)$, $\chi_A(t)=\sum(-1)^k e_k t^{n-k}$. Тогда:
    $$
    p_k - e_1 p_{k-1} + e_2 p_{k-2} - \dots + (-1)^{k-1}k e_k = 0,\quad k=1,\dots,n.
    $$
    Над полем характеристики $0$ знание $p_1,\dots,p_n$ восстанавливает $\chi_A$.
  \item[Когда использовать:] Восстановление $\chi_A$ по следам степеней. Критерий нильпотентности.
\end{description}

\subsection{Критерий нильпотентности через след}
\begin{description}
  \item[Формулировка:] Над полем характеристики $0$: $A$ нильпотентна $\iff$ $\operatorname{tr}(A^k)=0$ для $k=1,\dots,n$.
  \item[Когда использовать:] Доказательство нильпотентности, когда известны следы степеней.
  \item[Идея:] Все $p_k=0$ $\Rightarrow$ все $e_k=0$ $\Rightarrow$ $\chi_A(t)=t^n$ $\Rightarrow$ $A^n=0$.
\end{description}

\section{Ранг}

\subsection{Эквивалентные определения}
\begin{description}
  \item[Формулировка:] $\operatorname{rank}(A)=r$ $\iff$ максимальный размер ненулевого минора равен $r$ $\iff$ $A=\sum_{k=1}^r u_k v_k^T$ (минимальное представление) $\iff$ $\dim\operatorname{im}(A)=r$.
  \item[Когда использовать:] Любые задачи с рангом, связь с минорами.
\end{description}

\subsection{Неравенства Сильвестра и Фробениуса}
\begin{description}
  \item[Формулировка:] Сильвестр: $\operatorname{rank}(A)+\operatorname{rank}(B)-n\le\operatorname{rank}(AB)\le\min(\operatorname{rank}(A),\operatorname{rank}(B))$.

    Фробениус: $\operatorname{rank}(AB)+\operatorname{rank}(BC)\le\operatorname{rank}(B)+\operatorname{rank}(ABC)$.
  \item[Когда использовать:] Оценки ранга произведения, следствие: $AB=0\Rightarrow\operatorname{rank}(A)+\operatorname{rank}(B)\le n$.
  \item[Идея:] $\operatorname{rank}(AB)=\dim A(\operatorname{im}B)=\operatorname{rank}(B)-\dim(\operatorname{im}B\cap\ker A)$.
\end{description}

\section{Жорданова форма и спектральная теорема}

\subsection{Жорданова нормальная форма}
\begin{description}
  \item[Формулировка:] Над $\mathbb{C}$: $A=SJS^{-1}$, $J=\bigoplus J_{m_k}(\lambda_k)$, $J_m(\lambda)=\lambda I_m+N_m$. Размеры блоков однозначны. Число блоков размера $\ge k$ для $\lambda$ равно $\dim\ker(A-\lambda I)^k-\dim\ker(A-\lambda I)^{k-1}$.
  \item[Когда использовать:] Вычисление степеней $A^n$, $e^{tA}$, классификация нильпотентов.
\end{description}

\subsection{Спектральная теорема для нормальных матриц}
\begin{description}
  \item[Формулировка:] $A\in M_n(\mathbb{C})$ нормальна ($AA^*=A^*A$) $\iff$ существует унитарная $U$ с $U^*AU=\operatorname{diag}(\lambda_1,\dots,\lambda_n)$. Подслучаи:
    \begin{itemize}
      \item Эрмитова ($A=A^*$): $\lambda_i\in\mathbb{R}$.
      \item Унитарная ($A^*A=I$): $|\lambda_i|=1$.
      \item Вещественная симметричная: ортогонально диагонализуема над $\mathbb{R}$.
      \item Вещественная ортогональная: блочно-диагональная с $\pm1$ и $R_\theta$.
    \end{itemize}
  \item[Когда использовать:] Диагонализация эрмитовых/унитарных/нормальных матриц. Функциональное исчисление $f(A)=U f(\Lambda)U^*$.
  \item[Идея:] Шур $A=UTU^*$ $+$ условие нормальности $TT^*=T^*T$ для треугольной $T$ даёт $T$ диагональна.
\end{description}

\subsection{Триангуляризация Шура}
\begin{description}
  \item[Формулировка:] Над $\mathbb{C}$: существует унитарная $U$ и верхнетреугольная $T$ с $U^*AU=T$, $T_{ii}=\lambda_i$.
  \item[Когда использовать:] Универсальный инструмент: спектральный радиус, неравенство Шура~--Адамара, оценки степеней.
  \item[Идея:] Индукция: собственный вектор $\to$ ортонормированный базис $\to$ блочно-треугольная форма.
\end{description}

\section{Положительно (полу)определённые матрицы}

\subsection{Эквивалентные критерии PSD}
\begin{description}
  \item[Формулировка:] Для эрмитовой $A$ эквивалентны:
    \begin{enumerate}
      \item $x^*Ax\ge0$ для всех $x$;
      \item $\lambda_i\ge0$;
      \item $A=B^*B$ для некоторой $B$;
      \item все главные миноры $\ge0$.
    \end{enumerate}
    Для PD: строгие неравенства; в~(4) достаточно ведущих главных миноров (критерий Сильвестра).
  \item[Когда использовать:] Основной инструмент для работы с PSD-матрицами.
\end{description}

\subsection{Представление Грама}
\begin{description}
  \item[Формулировка:] $A\succeq0$ $\iff$ $A=B^*B$ $\iff$ $A_{ij}=\langle v_i,v_j\rangle$. $\operatorname{rank}(A)=\dim\operatorname{span}(v_i)$.
  \item[Когда использовать:] Сведение задач о PSD-матрицах к геометрии векторов.
\end{description}

\subsection{Критерий Сильвестра}
\begin{description}
  \item[Формулировка:] $A\succ0$ $\iff$ все ведущие главные миноры $\Delta_k>0$ ($k=1,\dots,n$). Для $A\succeq0$ нужны \textbf{все} главные миноры $\ge0$.
  \item[Когда использовать:] Проверка положительной определённости.
  \item[Идея:] $LDL^*$-разложение $+$ дополнение Шура.
\end{description}

\subsection{Разложение Холецкого}
\begin{description}
  \item[Формулировка:] $A\succ0$ $\iff$ существует единственная нижнетреугольная $L$ с $L_{ii}>0$ и $A=LL^*$.
  \item[Когда использовать:] Конструктивное представление PD-матрицы.
\end{description}

\subsection{Квадратный корень}
\begin{description}
  \item[Формулировка:] Для $A\succeq0$ существует единственный $A^{1/2}\succeq0$ с $(A^{1/2})^2=A$. Через спектр: $A^{1/2}=U\Lambda^{1/2}U^*$.
  \item[Когда использовать:] Подобие $AB\sim A^{1/2}BA^{1/2}$; функциональное исчисление.
\end{description}

\subsection{Дополнение Шура}
\begin{description}
  \item[Формулировка:] Для $M=\begin{pmatrix}A&B\\ B^*&D\end{pmatrix}$ с $D\succ0$: $M\succ0$ $\iff$ $A-BD^{-1}B^*\succ0$. Для $M\succeq0$: $D\succeq0$, $\operatorname{im}B\subseteq\operatorname{im}D$, $A-BD^+B^*\succeq0$.
  \item[Когда использовать:] Блочные PSD-критерии, конденсация сигнатуры.
\end{description}

\subsection{Порядок Лёвнера}
\begin{description}
  \item[Формулировка:] $A\succeq B$ $\iff$ $A-B\succeq0$. Свойства:
    \begin{itemize}
      \item $A\succeq B\Rightarrow C^*AC\succeq C^*BC$ (конгруэнтная инвариантность).
      \item $0\prec A\preceq B\Rightarrow B^{-1}\preceq A^{-1}$ (антитонность обратной).
      \item $0\preceq A\preceq B\Rightarrow A^{1/2}\preceq B^{1/2}$ (Лёвнера~--Хайнца).
      \item $A^2\succeq B^2$ \textbf{не следует} из $A\succeq B\succeq0$.
    \end{itemize}
  \item[Когда использовать:] Сравнение PSD-матриц без явных собственных чисел.
\end{description}

\subsection{Сингулярное разложение (SVD)}
\begin{description}
  \item[Формулировка:] Любая $A\in\mathbb{C}^{m\times n}$: $A=U\Sigma V^*$, $U,V$ унитарны, $\Sigma$ диагональна с $\sigma_1\ge\cdots\ge\sigma_r>0$. $\sigma_i=\sqrt{\lambda_i(A^*A)}$, $r=\operatorname{rank}(A)$.
  \item[Когда использовать:] Оценки норм, low-rank аппроксимация (Эккарт~--Янг), псевдообратная.
\end{description}

\subsection{Полярное разложение}
\begin{description}
  \item[Формулировка:] Любая $A$: $A=UP$, $U$ унитарна, $P=(A^*A)^{1/2}\succeq0$. Единственно при полном ранге.
  \item[Когда использовать:] Разделение вращения и растяжения.
\end{description}

\section{Евклидовы пространства и матрица Грама}

\subsection{Матрица Грама}
\begin{description}
  \item[Формулировка:] $G_{ij}=\langle v_i,v_j\rangle$. $G$ симметрична и PSD. $G\succ0$ $\iff$ $v_i$ линейно независимы. $\det G=(\operatorname{Vol})^2$, $\operatorname{rank}G=\dim\operatorname{span}(v_i)$. Любая PSD-матрица~--- матрица Грама некоторого набора векторов (с точностью до ортогонального преобразования).
  \item[Когда использовать:] Универсальный мост между геометрией и PSD-матрицами.
\end{description}

\subsection{Неравенство Адамара}
\begin{description}
  \item[Формулировка:] Для $A\in\mathbb{R}^{n\times n}$: $|\det A|\le\prod\|a_i\|$ (столбцы). Равенство $\iff$ столбцы ортогональны или есть нулевой. Для PSD: $\det H\le\prod H_{ii}$.
  \item[Когда использовать:] Верхние оценки определителя. Следствие: при $|a_{ij}|\le M$: $|\det A|\le M^n n^{n/2}$.
  \item[Идея:] QR-разложение: $|\det A|=\prod|R_{ii}|$, $|R_{ii}|\le\|a_i\|$.
\end{description}

\subsection{Ортогональная проекция}
\begin{description}
  \item[Формулировка:] Для $W\subseteq V$ существует $P_W$: $P_W^2=P_W$, $P_W^T=P_W$, $\operatorname{im}P_W=W$, $\ker P_W=W^\perp$. Для $x$: $\|x-P_Wx\|=\min_{y\in W}\|x-y\|$. Если столбцы $A$~--- базис $W$: $P_W=A(A^TA)^{-1}A^T$.
  \item[Когда использовать:] Минимизация расстояния, разложение $V=W\oplus W^\perp$.
\end{description}

\subsection{Неравенство Бесселя и равенство Парсеваля}
\begin{description}
  \item[Формулировка:] Для ортонормированной системы $e_1,\dots,e_k$: $\sum_{i=1}^k|\langle x,e_i\rangle|^2\le\|x\|^2$. Равенство $\iff$ $x\in\operatorname{span}(e_i)$. Для полного базиса~--- равенство Парсеваля.
  \item[Когда использовать:] Оценки сумм квадратов проекций.
\end{description}

\subsection{Неравенство Коши--Шварца}
\begin{description}
  \item[Формулировка:] $|\langle u,v\rangle|\le\|u\|\cdot\|v\|$, равенство $\iff$ $u,v$ линейно зависимы.
  \item[Эквивалент:] Матрица Грама $\begin{pmatrix}\|u\|^2 & \langle u,v\rangle\\ \langle v,u\rangle & \|v\|^2\end{pmatrix}$ PSD.
\end{description}

\subsection{Лемма Штейница}
\begin{description}
  \item[Формулировка:] $v_1,\dots,v_n\in\mathbb{R}^d$, $\|v_i\|\le1$, $\sum v_i=0$. Существует перестановка $\sigma$ с $\|v_{\sigma(1)}+\cdots+v_{\sigma(k)}\|\le d$ при всех $k$.
  \item[Когда использовать:] Задачи на упорядочивание сумм векторов.
\end{description}

\section{Билинейные и квадратичные формы. Сигнатура}

\subsection{Закон инерции Сильвестра}
\begin{description}
  \item[Формулировка:] Для симметричной $A\in\mathbb{R}^{n\times n}$ сигнатура $(p,q,r)$~--- числа положительных, отрицательных и нулевых собственных значений~--- инвариантна относительно конгруэнтности $A\mapsto C^TAC$.
  \item[Когда использовать:] Сравнение квадратичных форм в разных базисах.
  \item[Идея:] Максимальные подпространства, на которых $Q>0$ и $Q\le0$, имеют пересечение $\{0\}$.
\end{description}

\subsection{Сигнатура через спектр}
\begin{description}
  \item[Формулировка:] Для симметричной $A$: $p=\#\{\lambda_i>0\}$, $q=\#\{\lambda_i<0\}$, $r=\#\{\lambda_i=0\}$.
  \item[Когда использовать:] Быстрое определение сигнатуры.
\end{description}

\subsection{Формула Якоби}
\begin{description}
  \item[Формулировка:] Если все ведущие миноры $\Delta_k\neq0$, то существует верхнетреугольная $U$ с $U^TAU=\operatorname{diag}(\Delta_1,\Delta_2/\Delta_1,\dots,\Delta_n/\Delta_{n-1})$.
  \item[Когда использовать:] Вычисление сигнатуры по ведущим минорам.
\end{description}

\subsection{Одновременная диагонализация пары форм}
\begin{description}
  \item[Формулировка:] Если $A\succ0$ и $B$ симметрична, то существует $C$ с $C^TAC=I$, $C^TBC=\operatorname{diag}(\mu_i)$. Числа $\mu_i$~--- корни $\det(B-\mu A)=0$, все вещественны.
  \item[Когда использовать:] Сведение двух форм к каноническому виду.
  \item[Идея:] $A=LL^T$, $\widetilde B=L^{-1}BL^{-T}$ симметрична, ортогонально диагонализуема.
\end{description}

\subsection{Формула Хейнсворт (аддитивность сигнатуры)}
\begin{description}
  \item[Формулировка:] Для $M=\begin{pmatrix}A&B\\ B^T&C\end{pmatrix}$ с $A$ невырожденной: $\operatorname{In}(M)=\operatorname{In}(A)+\operatorname{In}(M/A)$, где $M/A=C-B^TA^{-1}B$.
  \item[Когда использовать:] Анализ сигнатуры блочных матриц.
\end{description}

\section{Определители}

\subsection{Вандермонд}
\begin{description}
  \item[Формулировка:] $\det(x_j^{i-1})=\prod_{i<j}(x_j-x_i)$. Обобщение: для монических многочленов $P_{i-1}$ степени $i-1$ определитель тот же.
  \item[Когда использовать:] Тождества с произвольными узлами. Редукция матриц вида $f(x_i,y_j)$.
\end{description}

\subsection{Блочные формулы и дополнение Шура}
\begin{description}
  \item[Формулировка:] $\det\begin{pmatrix}A&B\\ C&D\end{pmatrix}=\det A\cdot\det(D-CA^{-1}B)$ (при обратимой $A$). Если $CD=DC$, то $\det\begin{pmatrix}A&B\\ C&D\end{pmatrix}=\det(AD-BC)$.
  \item[Когда использовать:] Вычисление определителей блочных матриц.
\end{description}

\subsection{Коши--Бине}
\begin{description}
  \item[Формулировка:] Для $A\in M_{m\times n}$, $B\in M_{n\times m}$, $m\le n$:
    $$
    \det(AB)=\sum_{S,\,|S|=m}\det(A_{[m],S})\cdot\det(B_{S,[m]}).
    $$
  \item[Когда использовать:] Определитель Грама: $\det(A^TA)=\sum_S\det(A_{S})^2\ge0$.
  \item[Идея:] Внешняя степень: $\Lambda^m(AB)=\Lambda^m(A)\Lambda^m(B)$.
\end{description}

\subsection{Тождество Сильвестра}
\begin{description}
  \item[Формулировка:] $\det(I_m+AB)=\det(I_n+BA)$.
  \item[Когда использовать:] Переход между разными размерностями. Частный случай: $\det(I+uv^T)=1+v^Tu$.
\end{description}

\subsection{Возмущение ранга один}
\begin{description}
  \item[Формулировка:] $\det(A+uv^T)=\det(A)\cdot(1+v^TA^{-1}u)$ для обратимой $A$. Обобщение на ранг $k$: $\det(A+UV^T)=\det(A)\cdot\det(I_k+V^TA^{-1}U)$.
  \item[Когда использовать:] Матрицы вида $A+uv^T$, $aI+bJ$.
\end{description}

\subsection{Определитель Коши}
\begin{description}
  \item[Формулировка:] $\det(1/(a_i+b_j))=\dfrac{\prod_{i<j}(a_i-a_j)(b_i-b_j)}{\prod_{i,j}(a_i+b_j)}$.
  \item[Когда использовать:] Матрицы $1/(i+j-1)$ (Гильберт) и аналоги.
\end{description}

\subsection{Циркулянт}
\textit{Формулировка:} Циркулянт $C=\operatorname{circ}(c_0,\dots,c_{n-1})$ диагонализуется DFT: $\det C=\prod_{k=0}^{n-1} p(\omega^k)$, где $p(x)=\sum c_k x^k$, $\omega=e^{2\pi i/n}$.

\subsection{Трёхдиагональный определитель}
\begin{description}
  \item[Формулировка:] $D_n=a_n D_{n-1}-b_n c_{n-1} D_{n-2}$, $D_0=1$, $D_1=a_1$.
  \item[Когда использовать:] Матрицы с $a_{ij}=f(|i-j|)$, многочлены Чебышёва.
\end{description}

\subsection{Слитный Вандермонд}
\textit{Формулировка:} Для $\lambda_1,\dots,\lambda_r$ с кратностями $m_1+\cdots+m_r=n$ матрица, содержащая строки производных:
$$
\det V_\Lambda=\prod_{i<j}(\lambda_j-\lambda_i)^{m_i m_j}.
$$

\subsection{Неравенство Адамара (для строк)}
\textit{Формулировка:} $|\det A|\le\prod\|r_i\|$ (строки). Равенство $\iff$ строки ортогональны.

\medskip

\section{Специальные классы матриц}

\subsection{Нильпотентная матрица}
\begin{description}
  \item[Формулировка:] $N^k=0$. Эквивалентно: $\sigma(N)=\{0\}$, $\chi_N(t)=t^n$, $N$ подобна строго верхнетреугольной, $\operatorname{tr}(N^k)=0$ ($k=1,\dots,n$) над полем char $0$. Индекс нильпотентности $\le n$. Жорданова форма~--- блоки $J_{m_i}(0)$.
  \item[Ключевой факт:] $(I-N)^{-1}=I+N+\cdots+N^{n-1}$.
\end{description}

\subsection{Идемпотент}
\textit{Формулировка:} $P^2=P$. Тогда: $\mu_P\mid x(x-1)$ (простые корни) $\Rightarrow$ $P$ диагонализуема. $V=\operatorname{im}P\oplus\ker P$. $\sigma(P)\subseteq\{0,1\}$. $\operatorname{rank}(P)=\operatorname{tr}(P)$ (char $0$). $I-P$~--- тоже идемпотент, $\operatorname{im}(I-P)=\ker P$.

\subsection{Инволюция}
\textit{Формулировка:} $A^2=I$, char $\neq2$. Тогда: $\sigma(A)\subseteq\{\pm1\}$, $A$ диагонализуема. $V=V_+\oplus V_-$, $V_\pm=\ker(A\mp I)$. Проекторы $P_\pm=(I\pm A)/2$: $P_+P_-=0$, $P_++P_-=I$, $A=P_+-P_-$. $\operatorname{tr}(A)=\dim V_+-\dim V_-$, $\det A=(-1)^{\dim V_-}$.

\subsection{Коммутирующие матрицы}
\begin{description}
  \item[Формулировка:] Если $AB=BA$, то:
    \begin{itemize}
      \item Существует общий собственный вектор.
      \item Семейство попарно коммутирующих матриц одновременно триангуляризуемо (унитарно, над $\mathbb{C}$).
      \item Если все диагонализуемы~--- одновременно диагонализуемы.
      \item Спектр $p(A,B)=\{p(\lambda_i,\mu_i)\}$ с единой нумерацией (теорема Фробениуса).
      \item $K[A,B]$ (подалгебра, порождённая $A,B$) имеет $\dim\le n$ (Герстенхабер).
    \end{itemize}
  \item[Когда использовать:] Коммутирующие матрицы наследуют общий собственный базис.
\end{description}

\subsection{Уравнение Сильвестра}
\begin{description}
  \item[Формулировка:] $AX-XB=C$ имеет единственное решение $\iff$ $\sigma(A)\cap\sigma(B)=\emptyset$.
  \item[Когда использовать:] Блочная диагонализация, матричные уравнения.
\end{description}

\subsection{Коммутатор}
\textit{Формулировка:} $\operatorname{tr}(AB-BA)=0$. При char$=0$ уравнение $AB-BA=I$ неразрешимо в $M_n$.

\subsection{Лемма Джекобсона}
\textit{Формулировка:} $I-AB$ обратим $\iff$ $I-BA$ обратим, и $(I-BA)^{-1}=I+B(I-AB)^{-1}A$.

\medskip

\section{Произведение Кронекера}

\subsection{Базовая алгебра}
\textit{Формулировка:} $(A\otimes B)(C\otimes D)=(AC)\otimes(BD)$ (mixed product property). $(A\otimes B)^T=A^T\otimes B^T$, $(A\otimes B)^{-1}=A^{-1}\otimes B^{-1}$ (если обратимы). $A\otimes B$ перестановочно подобна $B\otimes A$.

\subsection{Спектр, след, определитель, ранг}
\textit{Формулировка:} $\sigma(A\otimes B)=\{\lambda_i\mu_j\}$. $\operatorname{tr}(A\otimes B)=\operatorname{tr}(A)\operatorname{tr}(B)$. $\det(A\otimes B)=(\det A)^m(\det B)^n$. $\operatorname{rank}(A\otimes B)=\operatorname{rank}(A)\cdot\operatorname{rank}(B)$.

\subsection{Векторизация}
\textit{Формулировка:} $\operatorname{vec}(AXB)=(B^T\otimes A)\operatorname{vec}(X)$. Частные случаи: $\operatorname{vec}(AX)=(I\otimes A)\operatorname{vec}(X)$, $\operatorname{vec}(XB)=(B^T\otimes I)\operatorname{vec}(X)$.

\subsection{Кронекерова сумма}
\textit{Формулировка:} $A\oplus B=A\otimes I_m+I_n\otimes B$. Спектр: $\{\lambda_i+\mu_j\}$. Теорема Сильвестра~--Розенблюма: $AX-XB=C$ разрешима единственно $\iff$ $\sigma(A)\cap\sigma(B)=\emptyset$.

\subsection{Теорема Шура (произведение Адамара)}
\textit{Формулировка:} Если $A,B\succeq0$, то $A\circ B\succeq0$ (поэлементное произведение). Главная подматрица $A\otimes B$.

\medskip

\section{Матрицы над $\mathbb{F}_p$}

\subsection{Тождество Фробениуса}
\begin{description}
  \item[Формулировка:] Для коммутирующих $A,B$ над полем char $p$: $(A+B)^{p^k}=A^{p^k}+B^{p^k}$.
  \item[Когда использовать:] Сокращение степеней в характеристике $p$.
\end{description}

\subsection{Порядок $\operatorname{GL}_n(\mathbb{F}_q)$}
\textit{Формулировка:} $|\operatorname{GL}_n(\mathbb{F}_q)|=\prod_{k=0}^{n-1}(q^n-q^k)$. Число $k$-мерных подпространств: $\binom{n}{k}_q$.

\subsection{Теорема Шевалле--Уорнинга}
\begin{description}
  \item[Формулировка:] $f_1,\dots,f_r\in\mathbb{F}_q[x_1,\dots,x_n]$, $\sum\deg f_i<n$. Число общих нулей в $\mathbb{F}_q^n$ делится на $p$. Если есть тривиальный ноль~--- есть и нетривиальный.
  \item[Когда использовать:] Комбинаторные задачи существования.
\end{description}

\subsection{Oddtown / Eventown (метод $\mathbb{F}_2$)}
\begin{description}
  \item[Формулировка:] Над $\mathbb{F}_2$ индикаторные векторы подмножеств: $\langle v_i,v_j\rangle\equiv|A_i\cap A_j|\pmod2$.
    \begin{itemize}
      \item Oddtown: $|A_i|$ нечётны, $|A_i\cap A_j|$ чётны ($i\neq j$) $\Rightarrow$ $m\le n$.
      \item Eventown: все $|A_i|$ и $|A_i\cap A_j|$ чётны $\Rightarrow$ $m\le2^{\lfloor n/2\rfloor}$.
    \end{itemize}
  \item[Идея:] Матрица Грама $VV^T=I_m$ над $\mathbb{F}_2$, $\operatorname{rank}\le n$.
\end{description}

\subsection{Комбинаторный Nullstellensatz Алона}
\begin{description}
  \item[Формулировка:] $f\in F[x_1,\dots,x_n]$, $\deg f=\sum t_i$, коэффициент при $\prod x_i^{t_i}$ ненулевой. Для $S_i\subseteq F$ с $|S_i|>t_i$ существует $s\in\prod S_i$ с $f(s)\neq0$.
  \item[Когда использовать:] Доказательство существования комбинаторных конфигураций.
\end{description}

\subsection{Характеристика 2: перманент = определитель}
\textit{Формулировка:} Над полем char $2$: $\operatorname{perm}(A)=\det(A)$, так как $\operatorname{sgn}(\sigma)\equiv1$.

\subsection{Симметричная матрица с нулевой диагональю в char 2}
\textit{Формулировка:} $A^T=A$, $A_{ii}=0$, char $2$ $\Rightarrow$ форма $x^TAy$ альтернированная, ранг чётен.

\medskip

\section{Матричная экспонента}

\subsection{Формула на жордановой форме}
\textit{Формулировка:} $e^{tJ_m(\lambda)}=e^{t\lambda}\sum_{k=0}^{m-1}\frac{t^k}{k!}N_m^k$. $e^{tA}=S\,e^{tJ}\,S^{-1}$.

\subsection{Спектральное отображение}
\begin{description}
  \item[Формулировка:] $\sigma(e^A)=e^{\sigma(A)}$ (с теми же кратностями). Размеры жордановых блоков сохраняются.
  \item[Следствие:] $\det e^A=e^{\operatorname{tr}A}$, $e^A$ всегда обратима.
\end{description}

\subsection{Условие $e^{A+B}=e^A e^B$}
\textit{Формулировка:} $AB=BA$ $\Rightarrow$ $e^{A+B}=e^A e^B$. Без коммутации равенство нарушается.

\subsection{Оценки $\|A^n\|$, $\|e^{tA}\|$}
\begin{description}
  \item[Формулировка:] $\|A^n\|\le C\cdot n^{m-1}\rho(A)^n$, $\|e^{tA}\|\le C(1+t^{m-1})e^{\alpha(A)t}$ ($t\ge0$), где $m$~--- макс.\ размер жорданова блока, $\alpha(A)=\max\operatorname{Re}\lambda$. Формула Гельфанда: $\rho(A)=\lim\|A^n\|^{1/n}$.
  \item[Когда использовать:] Асимптотика степеней/экспоненты. $A^k\to0$ $\iff$ $\rho(A)<1$.
\end{description}

\subsection{Образ $\exp$ на классических группах}
\textit{Формулировка:} $\exp:M_n(\mathbb{C})\to GL_n(\mathbb{C})$ сюръективно. $\exp:\{A^*=-A\}\to U(n)$ сюръективно. $\exp:\mathfrak{o}(n)\to SO(n)$ сюръективно. Над $\mathbb{R}$ не всякая вещественная матрица с $\det>0$ есть $e^B$.

\subsection{Уравнение Ляпунова}
\begin{description}
  \item[Формулировка:] Если $\alpha(A)<0$, то для $Q\succ0$: $P=\int_0^\infty e^{tA^*}Q e^{tA}\,dt\succ0$ и $A^*P+PA=-Q$. Обратно: $P\succ0$, $A^*P+PA\prec0$ $\Rightarrow$ $\alpha(A)<0$.
  \item[Когда использовать:] Критерии устойчивости.
\end{description}

\subsection{Формула Ли--Троттера}
\textit{Формулировка:} $e^{A+B}=\lim_{n\to\infty}(e^{A/n}e^{B/n})^n$.

\subsection{Преобразование Кэли}
\textit{Формулировка:} $\Phi(A)=(I-A)(I+A)^{-1}$. $A^*=-A$ $\iff$ $\Phi(A)$ унитарна. $A^T=-A$ $\iff$ $\Phi(A)\in SO(n)$.

\medskip

\section{Целочисленные матрицы (Smith normal form)}

\subsection{Нормальная форма Смита}
\textit{Формулировка:} Для $A\in M_{m\times n}(\mathbb{Z})$ ранга $r$ существуют унимодулярные $U,V$ с $UAV=\operatorname{diag}(d_1,\dots,d_r,0,\dots)$, $d_1\mid d_2\mid\cdots\mid d_r$, $d_i>0$. Числа $d_i$~--- инвариантные факторы. Для квадратной $A$: $\prod d_i=|\det A|$.

\subsection{Инвариантные факторы через миноры}
\begin{description}
  \item[Формулировка:] $\Delta_k(A)=\gcd$ всех $k\times k$ миноров $A$. $d_k=\Delta_k/\Delta_{k-1}$.
  \item[Когда использовать:] Вычисление Смита без приведения матрицы.
\end{description}

\subsection{Структура коядра}
\textit{Формулировка:} $\mathbb{Z}^n/A\mathbb{Z}^n\cong\bigoplus\mathbb{Z}/d_i\oplus\mathbb{Z}^{n-r}$. Для невырожденной $A$: $|\mathbb{Z}^n/A\mathbb{Z}^n|=|\det A|$.

\subsection{Нормальная форма Эрмита}
\textit{Формулировка:} Для $A$ полного строчного ранга существует унимодулярная $U$ с $UA=H$, где $H$ верхнетреугольная с $h_{ii}>0$, $0\le h_{ij}<h_{ii}$ при $j<i$.

\subsection{Редукция по простому модулю}
\begin{description}
  \item[Формулировка:] $A\mapsto A\bmod p$~--- гомоморфизм $M_n(\mathbb{Z})\to M_n(\mathbb{F}_p)$. Сохраняет $\det$, сумму, произведение.
  \item[Когда использовать:] Перенос ограничений с $\mathbb{Z}$ на $\mathbb{F}_p$.
\end{description}

\medskip

\section{Минимаксные характеризации и неравенства}

\subsection{Куранта--Фишера}
\begin{description}
  \item[Формулировка:] Для эрмитовой $A$ с $\lambda_1\ge\cdots\ge\lambda_n$:
    $$
    \lambda_k=\max_{\dim V=k}\min_{x\in V,\|x\|=1}x^*Ax=\min_{\dim V=n-k+1}\max_{x\in V,\|x\|=1}x^*Ax.
    $$
    Для сингулярных чисел: $\sigma_k(A)=\max_{\dim V=k}\min_{x\in V,\|x\|=1}\|Ax\|$.
  \item[Когда использовать:] Оценки экстремальных собственных значений.
\end{description}

\subsection{Перемежение Коши}
\textit{Формулировка:} Для главной подматрицы $B$ размера $n-1$: $\lambda_1\ge\mu_1\ge\lambda_2\ge\mu_2\ge\cdots\ge\mu_{n-1}\ge\lambda_n$.

\subsection{Неравенства Вейля}
\textit{Формулировка:} Для эрмитовых $A,B$: $\lambda_{i+j-1}(A+B)\le\lambda_i(A)+\lambda_j(B)$, $i+j-1\le n$. Для сингулярных чисел: $\sigma_{i+j-1}(A+B)\le\sigma_i(A)+\sigma_j(B)$.

\subsection{Неравенство Шура~--Адамара}
\textit{Формулировка:} $\sum|\lambda_i|^2\le\|A\|_F^2$. Равенство $\iff$ $A$ нормальна.

\subsection{Неравенство Минковского}
\textit{Формулировка:} Для $A,B\succeq0$: $\det(A+B)^{1/n}\ge\det(A)^{1/n}+\det(B)^{1/n}$. Равенство $\iff$ $A,B$ пропорциональны.

\subsection{Неравенство Фишера}
\textit{Формулировка:} Для PSD-блочной $H=\begin{pmatrix}A&B\\ B^T&D\end{pmatrix}$: $\det H\le\det A\cdot\det D$. Равенство $\iff$ $B=0$.
